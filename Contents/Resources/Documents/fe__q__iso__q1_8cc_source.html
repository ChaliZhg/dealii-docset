103 template <int dim, int spacedim>
104 FiniteElementDomination::Domination
105 FE_Q_iso_Q1<dim,spacedim>::
106 compare_for_face_domination (const FiniteElement<dim,spacedim> &fe_other) const
107 {
108  if (const FE_Q_iso_Q1<dim,spacedim> *fe_q_iso_q1_other
109  = dynamic_cast<const FE_Q_iso_Q1<dim,spacedim>*>(&fe_other))
110  {
111  // different behavior as in FE_Q: as FE_Q_iso_Q1(2) is not a subspace of
112  // FE_Q_iso_Q1(3), need that the element degrees are multiples of each
113  // other
114  if (this->degree < fe_q_iso_q1_other->degree &&
115  fe_q_iso_q1_other->degree % this->degree == 0)
116  return FiniteElementDomination::this_element_dominates;
117  else if (this->degree == fe_q_iso_q1_other->degree)
118  return FiniteElementDomination::either_element_can_dominate;
119  else if (this->degree > fe_q_iso_q1_other->degree &&
120  this->degree % fe_q_iso_q1_other->degree == 0)
121  return FiniteElementDomination::other_element_dominates;
122  else
123  return FiniteElementDomination::neither_element_dominates;
124  }
125  else if (const FE_Nothing<dim> *fe_nothing = dynamic_cast<const FE_Nothing<dim>*>(&fe_other))
126  {
127  if (fe_nothing->is_dominating())
128  {
129  return FiniteElementDomination::other_element_dominates;
130  }
131  else
132  {
133  // the FE_Nothing has no degrees of freedom and it is typically used in
134  // a context where we don't require any continuity along the interface
135  return FiniteElementDomination::no_requirements;
136  }
137  }
138 
139  Assert (false, ExcNotImplemented());
140  return FiniteElementDomination::neither_element_dominates;
141 }
