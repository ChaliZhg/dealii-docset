1430 inline
1431 void
1432 ConstraintMatrix::add_line (const size_type line)
1433 {
1434  Assert (sorted==false, ExcMatrixIsClosed());
1435 
1436  // the following can happen when we compute with distributed meshes and dof
1437  // handlers and we constrain a degree of freedom whose number we don't have
1438  // locally. if we don't abort here the program will try to allocate several
1439  // terabytes of memory to resize the various arrays below :-)
1440  Assert (line != numbers::invalid_size_type,
1441  ExcInternalError());
1442  const size_type line_index = calculate_line_index (line);
1443 
1444  // check whether line already exists; it may, in which case we can just quit
1445  if (is_constrained(line))
1446  return;
1447 
1448  // if necessary enlarge vector of existing entries for cache
1449  if (line_index >= lines_cache.size())
1450  lines_cache.resize (std::max(2*static_cast<size_type>(lines_cache.size()),
1451  line_index+1),
1452  numbers::invalid_size_type);
1453 
1454  // push a new line to the end of the list
1455  lines.emplace_back ();
1456  lines.back().line = line;
1457  lines.back().inhomogeneity = 0.;
1458  lines_cache[line_index] = lines.size()-1;
1459 }

1463 inline
1464 void
1465 ConstraintMatrix::add_entry (const size_type line,
1466  const size_type column,
1467  const double value)
1468 {
1469  Assert (sorted==false, ExcMatrixIsClosed());
1470  Assert (line != column,
1471  ExcMessage ("Can't constrain a degree of freedom to itself"));
1472 
1473  // Ensure that the current line is present in the cache:
1474  const size_type line_index = calculate_line_index(line);
1475  Assert (line_index < lines_cache.size(),
1476  ExcMessage("The current ConstraintMatrix does not contain the line "
1477  "for the current entry. Call ConstraintMatrix::add_line "
1478  "before calling this function."));
1479 
1480  // if in debug mode, check whether an entry for this column already exists
1481  // and if it's the same as the one entered at present
1482  //
1483  // in any case: exit the function if an entry for this column already
1484  // exists, since we don't want to enter it twice
1485  Assert (lines_cache[line_index] != numbers::invalid_size_type,
1486  ExcInternalError());
1487  Assert (!local_lines.size() || local_lines.is_element(column),
1488  ExcColumnNotStoredHere(line, column));
1489  ConstraintLine *line_ptr = &lines[lines_cache[line_index]];
1490  Assert (line_ptr->line == line, ExcInternalError());
1491  for (ConstraintLine::Entries::const_iterator
1492  p=line_ptr->entries.begin();
1493  p != line_ptr->entries.end(); ++p)
1494  if (p->first == column)
1495  {
1496  Assert (std::fabs(p->second - value) < 1.e-14,
1497  ExcEntryAlreadyExists(line, column, p->second, value));
1498  return;
1499  }
1500 
1501  line_ptr->entries.emplace_back (column, value);
1502 }

1626 template <class VectorType>
1627 inline
1628 void ConstraintMatrix::distribute_local_to_global (
1629  const size_type index,
1630  const double value,
1631  VectorType &global_vector) const
1632 {
1633  Assert (lines.empty() || sorted == true, ExcMatrixNotClosed());
1634 
1635  if (is_constrained(index) == false)
1636  global_vector(index) += value;
1637  else
1638  {
1639  const ConstraintLine &position =
1640  lines[lines_cache[calculate_line_index(index)]];
1641  for (size_type j=0; j<position.entries.size(); ++j)
1642  global_vector(position.entries[j].first)
1643  += value * position.entries[j].second;
1644  }
1645 }

1664  else
