288 template <int dim>
289 template <typename Number>
290 inline
291 SymmetricTensor<4, dim, Number>
292 Physics::Elasticity::StandardTensors<dim>::Dev_P (const Tensor<2, dim, Number> &F)
293 {
294  const Number det_F = determinant(F);
295  Assert(det_F > Number(0.0),
296  ExcMessage("Deformation gradient has a negative determinant."));
297  const Tensor<2,dim,Number> C_ns = transpose(F)*F;
298  const SymmetricTensor<2,dim,Number> C = symmetrize(C_ns);
299  const SymmetricTensor<2,dim,Number> C_inv = symmetrize(invert(C_ns));
300 
301  // See Wriggers p46 equ 3.125 (but transpose indices)
302  SymmetricTensor<4,dim,Number> Dev_P = outer_product(C,C_inv); // Dev_P = C_x_C_inv
303  Dev_P /= -dim; // Dev_P = -[1/dim]C_x_C_inv
304  Dev_P += SymmetricTensor<4,dim,Number>(S); // Dev_P = S - [1/dim]C_x_C_inv
305  Dev_P *= std::pow(det_F, -2.0/dim); // Dev_P = J^{-2/dim} [S - [1/dim]C_x_C_inv]
306 
307  return Dev_P;
308 }

313 template <typename Number>
314 inline
315 SymmetricTensor<4, dim, Number>
316 Physics::Elasticity::StandardTensors<dim>::Dev_P_T (const Tensor<2, dim, Number> &F)
317 {
318  const Number det_F = determinant(F);
319  Assert(det_F > Number(0.0),
320  ExcMessage("Deformation gradient has a negative determinant."));
321  const Tensor<2,dim,Number> C_ns = transpose(F)*F;
322  const SymmetricTensor<2,dim,Number> C = symmetrize(C_ns);
323  const SymmetricTensor<2,dim,Number> C_inv = symmetrize(invert(C_ns));
324 
325  // See Wriggers p46 equ 3.125 (not transposed)
326  SymmetricTensor<4,dim,Number> Dev_P_T = outer_product(C_inv,C); // Dev_P = C_inv_x_C
327  Dev_P_T /= -dim; // Dev_P = -[1/dim]C_inv_x_C
328  Dev_P_T += SymmetricTensor<4,dim,Number>(S); // Dev_P = S - [1/dim]C_inv_x_C
329  Dev_P_T *= std::pow(det_F, -2.0/dim); // Dev_P = J^{-2/dim} [S - [1/dim]C_inv_x_C]
330 
331  return Dev_P_T;
332 }

347 template <int dim>
348 template <typename Number>
349 inline
350 SymmetricTensor<4, dim, Number>
351 Physics::Elasticity::StandardTensors<dim>::dC_inv_dC (const Tensor<2, dim, Number> &F)
352 {
353  const SymmetricTensor<2,dim,Number> C_inv = symmetrize(invert(transpose(F)*F));
354 
355  SymmetricTensor<4,dim,Number> dC_inv_dC;
356  for (unsigned int A=0; A<dim; ++A)
357  for (unsigned int B=A; B<dim; ++B)
358  for (unsigned int C=0; C<dim; ++C)
359  for (unsigned int D=C; D<dim; ++D)
360  dC_inv_dC[A][B][C][D] -= 0.5*(C_inv[A][C] * C_inv[B][D] + C_inv[A][D] * C_inv[B][C] );
361 
362  return dC_inv_dC;
363 }
