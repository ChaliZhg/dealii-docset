105  void
106  Partitioner::set_owned_indices (const IndexSet &locally_owned_indices)
107  {
108  if (Utilities::System::job_supports_mpi() == true)
109  {
110  my_pid = Utilities::MPI::this_mpi_process(communicator);
111  n_procs = Utilities::MPI::n_mpi_processes(communicator);
112  }
113  else
114  {
115  my_pid = 0;
116  n_procs = 1;
117  }
118 
119  // set the local range
120  Assert (locally_owned_indices.is_contiguous() == true,
121  ExcMessage ("The index set specified in locally_owned_indices "
122  "is not contiguous."));
123  locally_owned_indices.compress();
124  if (locally_owned_indices.n_elements()>0)
125  local_range_data = std::pair<types::global_dof_index, types::global_dof_index>
126  (locally_owned_indices.nth_index_in_set(0),
127  locally_owned_indices.nth_index_in_set(0) +
128  locally_owned_indices.n_elements());
129  AssertThrow (local_range_data.second-local_range_data.first <
130  static_cast<types::global_dof_index>(std::numeric_limits<unsigned int>::max()),
131  ExcMessage("Index overflow: This class supports at most 2^32-1 locally owned vector entries"));
132  locally_owned_range_data.set_size (locally_owned_indices.size());
133  locally_owned_range_data.add_range (local_range_data.first, local_range_data.second);
134  locally_owned_range_data.compress();
135 
136  ghost_indices_data.set_size (locally_owned_indices.size());
137  }

141  void
142  Partitioner::set_ghost_indices (const IndexSet &ghost_indices_in)
143  {
144  // Set ghost indices from input. To be sure that no entries from the
145  // locally owned range are present, subtract the locally owned indices
146  // in any case.
147  Assert (ghost_indices_in.n_elements() == 0 ||
148  ghost_indices_in.size() == locally_owned_range_data.size(),
149  ExcDimensionMismatch (ghost_indices_in.size(),
150  locally_owned_range_data.size()));
151 
152  ghost_indices_data = ghost_indices_in;
153  if (ghost_indices_data.size() != locally_owned_range_data.size())
154  ghost_indices_data.set_size(locally_owned_range_data.size());
155  ghost_indices_data.subtract_set (locally_owned_range_data);
156  ghost_indices_data.compress();
157  AssertThrow (ghost_indices_data.n_elements() <
158  static_cast<types::global_dof_index>(std::numeric_limits<unsigned int>::max()),
159  ExcMessage("Index overflow: This class supports at most 2^32-1 ghost elements"));
160  n_ghost_indices_data = ghost_indices_data.n_elements();
161 
162  have_ghost_indices =
163  Utilities::MPI::sum(n_ghost_indices_data, communicator) > 0;
164 
165  // In the rest of this function, we determine the point-to-point
166  // communication pattern of the partitioner. We make up a list with both
167  // the processors the ghost indices actually belong to, and the indices
168  // that are locally held but ghost indices of other processors. This
169  // allows then to import and export data very easily.
170 
171  // find out the end index for each processor and communicate it (this
172  // implies the start index for the next processor)
173 #ifdef DEAL_II_WITH_MPI
174  if (n_procs < 2)
175  {
176  Assert (ghost_indices_data.n_elements() == 0, ExcInternalError());
177  Assert (n_import_indices_data == 0, ExcInternalError());
178  Assert (n_ghost_indices_data == 0, ExcInternalError());
179  return;
180  }
181 
182  std::vector<types::global_dof_index> first_index (n_procs+1);
183  // Allow non-zero start index for the vector. send this data to all
184  // processors
185  first_index[0] = local_range_data.first;
186  int ierr = MPI_Bcast(&first_index[0], 1, DEAL_II_DOF_INDEX_MPI_TYPE,
187  0, communicator);
188  AssertThrowMPI(ierr);
189 
190  // Get the end-of-local_range for all processors
191  ierr = MPI_Allgather(&local_range_data.second, 1,
192  DEAL_II_DOF_INDEX_MPI_TYPE, &first_index[1], 1,
193  DEAL_II_DOF_INDEX_MPI_TYPE, communicator);
194  AssertThrowMPI(ierr);
195  first_index[n_procs] = global_size;
196 
197  // fix case when there are some processors without any locally owned
198  // indices: then there might be a zero in some entries
199  if (global_size > 0)
200  {
201  unsigned int first_proc_with_nonzero_dofs = 0;
202  for (unsigned int i=0; i<n_procs; ++i)
203  if (first_index[i+1]>0)
204  {
205  first_proc_with_nonzero_dofs = i;
206  break;
207  }
208  for (unsigned int i=first_proc_with_nonzero_dofs+1; i<n_procs; ++i)
209  if (first_index[i] == 0)
210  first_index[i] = first_index[i-1];
211 
212  // correct if our processor has a wrong local range
213  if (first_index[my_pid] != local_range_data.first)
214  {
215  Assert(local_range_data.first == local_range_data.second,
216  ExcInternalError());
217  local_range_data.first = local_range_data.second = first_index[my_pid];
218  }
219  }
220 
221  // Allocate memory for data that will be exported
222  std::vector<types::global_dof_index> expanded_ghost_indices (n_ghost_indices_data);
223  unsigned int n_ghost_targets = 0;
224  if (n_ghost_indices_data > 0)
225  {
226  // Create first a vector of ghost_targets from the list of ghost
227  // indices and then push back new values. When we are done, copy the
228  // data to that field of the partitioner. This way, the variable
229  // ghost_targets will have exactly the size we need, whereas the
230  // vector filled with emplace_back might actually be too long.
231  unsigned int current_proc = 0;
232  ghost_indices_data.fill_index_vector (expanded_ghost_indices);
233  types::global_dof_index current_index = expanded_ghost_indices[0];
234  while (current_index >= first_index[current_proc+1])
235  current_proc++;
236  std::vector<std::pair<unsigned int, unsigned int> > ghost_targets_temp
237  (1, std::pair<unsigned int, unsigned int>(current_proc, 0));
238  n_ghost_targets++;
239 
240  for (unsigned int iterator=1; iterator<n_ghost_indices_data; ++iterator)
241  {
242  current_index = expanded_ghost_indices[iterator];
243  while (current_index >= first_index[current_proc+1])
244  current_proc++;
245  AssertIndexRange (current_proc, n_procs);
246  if ( ghost_targets_temp[n_ghost_targets-1].first < current_proc)
247  {
248  ghost_targets_temp[n_ghost_targets-1].second =
249  iterator - ghost_targets_temp[n_ghost_targets-1].second;
250  ghost_targets_temp.emplace_back (current_proc, iterator);
251  n_ghost_targets++;
252  }
253  }
254  ghost_targets_temp[n_ghost_targets-1].second =
255  n_ghost_indices_data - ghost_targets_temp[n_ghost_targets-1].second;
256  ghost_targets_data = ghost_targets_temp;
257  }
258  // find the processes that want to import to me
259  {
260  std::vector<int> send_buffer (n_procs, 0);
261  std::vector<int> receive_buffer (n_procs, 0);
262  for (unsigned int i=0; i<n_ghost_targets; i++)
263  send_buffer[ghost_targets_data[i].first] = ghost_targets_data[i].second;
264 
265  const int ierr = MPI_Alltoall (&send_buffer[0], 1, MPI_INT, &receive_buffer[0], 1,
266  MPI_INT, communicator);
267  AssertThrowMPI(ierr);
268 
269  // allocate memory for import data
270  std::vector<std::pair<unsigned int,unsigned int> > import_targets_temp;
271  n_import_indices_data = 0;
272  for (unsigned int i=0; i<n_procs; i++)
273  if (receive_buffer[i] > 0)
274  {
275  n_import_indices_data += receive_buffer[i];
276  import_targets_temp.emplace_back(i, receive_buffer[i]);
277  }
278  // copy, don't move, to get deterministic memory usage.
279  import_targets_data = import_targets_temp;
280  }
281 
282  // send and receive indices for import data. non-blocking receives and
283  // blocking sends
284  std::vector<types::global_dof_index> expanded_import_indices (n_import_indices_data);
285  {
286  unsigned int current_index_start = 0;
287  std::vector<MPI_Request> import_requests (import_targets_data.size());
288  for (unsigned int i=0; i<import_targets_data.size(); i++)
289  {
290  const int ierr = MPI_Irecv (&expanded_import_indices[current_index_start],
291  import_targets_data[i].second,
292  DEAL_II_DOF_INDEX_MPI_TYPE,
293  import_targets_data[i].first,
294  import_targets_data[i].first,
295  communicator, &import_requests[i]);
296  AssertThrowMPI(ierr);
297  current_index_start += import_targets_data[i].second;
298  }
299  AssertDimension (current_index_start, n_import_indices_data);
300 
301  // use blocking send
302  current_index_start = 0;
303  for (unsigned int i=0; i<n_ghost_targets; i++)
304  {
305  const int ierr = MPI_Send (&expanded_ghost_indices[current_index_start],
306  ghost_targets_data[i].second, DEAL_II_DOF_INDEX_MPI_TYPE,
307  ghost_targets_data[i].first, my_pid,
308  communicator);
309  AssertThrowMPI(ierr);
310  current_index_start += ghost_targets_data[i].second;
311  }
312  AssertDimension (current_index_start, n_ghost_indices_data);
313 
314  if (import_requests.size()>0)
315  {
316  const int ierr = MPI_Waitall (import_requests.size(),
317  &import_requests[0],
318  MPI_STATUSES_IGNORE);
319  AssertThrowMPI(ierr);
320  }
321 
322  // transform import indices to local index space and compress
323  // contiguous indices in form of ranges
324  {
325  types::global_dof_index last_index = numbers::invalid_dof_index-1;
326  std::vector<std::pair<unsigned int,unsigned int> > compressed_import_indices;
327  for (unsigned int i=0; i<n_import_indices_data; i++)
328  {
329  Assert (expanded_import_indices[i] >= local_range_data.first &&
330  expanded_import_indices[i] < local_range_data.second,
331  ExcIndexRange(expanded_import_indices[i], local_range_data.first,
332  local_range_data.second));
333  types::global_dof_index new_index = (expanded_import_indices[i] -
334  local_range_data.first);
335  Assert(new_index<numbers::invalid_unsigned_int,
336  ExcNotImplemented());
337  if (new_index == last_index+1)
338  compressed_import_indices.back().second++;
339  else
340  {
341  compressed_import_indices.emplace_back (new_index,new_index+1);
342  }
343  last_index = new_index;
344  }
345  import_indices_data = compressed_import_indices;
346 
347  // sanity check
348 #ifdef DEBUG
349  const types::global_dof_index n_local_dofs = local_range_data.second-local_range_data.first;
350  for (unsigned int i=0; i<import_indices_data.size(); ++i)
351  {
352  AssertIndexRange (import_indices_data[i].first, n_local_dofs);
353  AssertIndexRange (import_indices_data[i].second-1, n_local_dofs);
354  }
355 #endif
356  }
357  }
358 #endif

363  bool
364  Partitioner::is_compatible (const Partitioner &part) const
365  {
366  // if the partitioner points to the same memory location as the calling
367  // processor
368  if (&part == this)
369  return true;
370 #ifdef DEAL_II_WITH_MPI
371  if (Utilities::MPI::job_supports_mpi())
372  {
373  int communicators_same = 0;
374  const int ierr = MPI_Comm_compare (part.communicator, communicator,
375  &communicators_same);
376  AssertThrowMPI(ierr);
377  if (!(communicators_same == MPI_IDENT ||
378  communicators_same == MPI_CONGRUENT))
379  return false;
380  }
381 #endif
382  return (global_size == part.global_size &&
383  local_range_data == part.local_range_data &&
384  ghost_indices_data == part.ghost_indices_data);
385  }
