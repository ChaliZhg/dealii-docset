420  LocalResults<number>::initialize_matrices
421  (const MatrixBlockVector<MatrixType> &matrices,
422  bool both)
423  {
424  M1.resize(matrices.size());
425  if (both)
426  M2.resize(matrices.size());
427  for (unsigned int i=0; i<matrices.size(); ++i)
428  {
429  const unsigned int row = matrices.block(i).row;
430  const unsigned int col = matrices.block(i).column;
431 
432  M1[i].row = row;
433  M1[i].column = col;
434  if (both)
435  {
436  M2[i].row = row;
437  M2[i].column = col;
438  }
439  }

657  {

659  << " " << M1[i].matrix.m() << 'x' << M1[i].matrix.n();
660  if (i < M2.size())
