1932  inline
1933  void
1934  PreconditionBase::transpose ()
1935  {
1936  // This only flips a flag that tells
1937  // Trilinos that any vmult operation
1938  // should be done with the
1939  // transpose. However, the matrix
1940  // structure is not reset.
1941  int ierr;
1942 
1943  if (!preconditioner->UseTranspose())
1944  {
1945  ierr = preconditioner->SetUseTranspose (true);
1946  AssertThrow (ierr == 0, ExcTrilinosError(ierr));
1947  }
1948  else
1949  {
1950  ierr = preconditioner->SetUseTranspose (false);
1951  AssertThrow (ierr == 0, ExcTrilinosError(ierr));
1952  }
1953  }

1989  // For the implementation of the <code>vmult</code> function with deal.II
1990  // data structures we note that invoking a call of the Trilinos
1991  // preconditioner requires us to use Epetra vectors as well. We do this by
1992  // providing a view, i.e., feed Trilinos with a pointer to the data, so we
1993  // avoid copying the content of the vectors during the iteration (this
1994  // function is only useful when used in serial anyway). In the declaration
1995  // of the right hand side, we need to cast the source vector (that is
1996  // <code>const</code> in all deal.II calls) to non-constant value, as this
1997  // is the way Trilinos wants to have them.
1998  inline
1999  void PreconditionBase::vmult (::Vector<double> &dst,
2000  const ::Vector<double> &src) const
2001  {
2002  AssertDimension (static_cast<TrilinosWrappers::types::int_type>(dst.size()),
2003  preconditioner->OperatorDomainMap().NumMyElements());
2004  AssertDimension (static_cast<TrilinosWrappers::types::int_type>(src.size()),
2005  preconditioner->OperatorRangeMap().NumMyElements());
2006  Epetra_Vector tril_dst (View, preconditioner->OperatorDomainMap(),
2007  dst.begin());
2008  Epetra_Vector tril_src (View, preconditioner->OperatorRangeMap(),
2009  const_cast<double *>(src.begin()));
2010 
2011  const int ierr = preconditioner->ApplyInverse (tril_src, tril_dst);
2012  AssertThrow (ierr == 0, ExcTrilinosError(ierr));
2013  }
