129 template <typename VectorType>
130 template <int dim, int spacedim>
131 void MGTransferPrebuilt<VectorType>::build_matrices
132 (const DoFHandler<dim,spacedim> &mg_dof)
133 {
134  const unsigned int n_levels = mg_dof.get_triangulation().n_global_levels();
135  const unsigned int dofs_per_cell = mg_dof.get_fe().dofs_per_cell;
136 
137  this->sizes.resize(n_levels);
138  for (unsigned int l=0; l<n_levels; ++l)
139  this->sizes[l] = mg_dof.n_dofs(l);
140 
141  // reset the size of the array of
142  // matrices. call resize(0) first,
143  // in order to delete all elements
144  // and clear their memory. then
145  // repopulate these arrays
146  //
147  // note that on resize(0), the
148  // shared_ptr class takes care of
149  // deleting the object it points to
150  // by itself
151  prolongation_matrices.resize (0);
152  prolongation_sparsities.resize (0);
153  prolongation_matrices.reserve (n_levels - 1);
154  prolongation_sparsities.reserve (n_levels - 1);
155 
156  for (unsigned int i=0; i<n_levels-1; ++i)
157  {
158  prolongation_sparsities.emplace_back
159  (new typename internal::MatrixSelector<VectorType>::Sparsity);
160  prolongation_matrices.emplace_back
161  (new typename internal::MatrixSelector<VectorType>::Matrix);
162  }
163 
164  // two fields which will store the
165  // indices of the multigrid dofs
166  // for a cell and one of its children
167  std::vector<types::global_dof_index> dof_indices_parent (dofs_per_cell);
168  std::vector<types::global_dof_index> dof_indices_child (dofs_per_cell);
169  std::vector<types::global_dof_index> entries (dofs_per_cell);
170 
171  // for each level: first build the sparsity
172  // pattern of the matrices and then build the
173  // matrices themselves. note that we only
174  // need to take care of cells on the coarser
175  // level which have children
176  for (unsigned int level=0; level<n_levels-1; ++level)
177  {
178  // reset the dimension of the structure. note that for the number of
179  // entries per row, the number of parent dofs coupling to a child dof is
180  // necessary. this, of course, is the number of degrees of freedom per
181  // cell
182  //
183  // increment dofs_per_cell since a useless diagonal element will be
184  // stored
185  IndexSet level_p1_relevant_dofs;
186  DoFTools::extract_locally_relevant_level_dofs(mg_dof, level+1,
187  level_p1_relevant_dofs);
188  DynamicSparsityPattern dsp (this->sizes[level+1],
189  this->sizes[level],
190  level_p1_relevant_dofs);
191  typename DoFHandler<dim>::cell_iterator cell, endc = mg_dof.end(level);
192  for (cell=mg_dof.begin(level); cell != endc; ++cell)
193  if (cell->has_children() &&
194  ( mg_dof.get_triangulation().locally_owned_subdomain()==numbers::invalid_subdomain_id
195  || cell->level_subdomain_id()==mg_dof.get_triangulation().locally_owned_subdomain()
196  ))
197  {
198  cell->get_mg_dof_indices (dof_indices_parent);
199 
200  Assert(cell->n_children()==GeometryInfo<dim>::max_children_per_cell,
201  ExcNotImplemented());
202  for (unsigned int child=0; child<cell->n_children(); ++child)
203  {
204  // set an alias to the prolongation matrix for this child
205  const FullMatrix<double> &prolongation
206  = mg_dof.get_fe().get_prolongation_matrix (child,
207  cell->refinement_case());
208 
209  Assert (prolongation.n() != 0, ExcNoProlongation());
210 
211  cell->child(child)->get_mg_dof_indices (dof_indices_child);
212 
213  // now tag the entries in the
214  // matrix which will be used
215  // for this pair of parent/child
216  for (unsigned int i=0; i<dofs_per_cell; ++i)
217  {
218  entries.resize(0);
219  for (unsigned int j=0; j<dofs_per_cell; ++j)
220  if (prolongation(i,j) != 0)
221  entries.push_back (dof_indices_parent[j]);
222  dsp.add_entries (dof_indices_child[i],
223  entries.begin(), entries.end());
224  }
225  }
226  }
227 
228  internal::MatrixSelector<VectorType>::reinit(*prolongation_matrices[level],
229  *prolongation_sparsities[level],
230  level,
231  dsp,
232  mg_dof);
233  dsp.reinit(0,0);
234 
235  FullMatrix<double> prolongation;
236 
237  // now actually build the matrices
238  for (cell=mg_dof.begin(level); cell != endc; ++cell)
239  if (cell->has_children() &&
240  (mg_dof.get_triangulation().locally_owned_subdomain()==numbers::invalid_subdomain_id
241  || cell->level_subdomain_id()==mg_dof.get_triangulation().locally_owned_subdomain())
242  )
243  {
244  cell->get_mg_dof_indices (dof_indices_parent);
245 
246  Assert(cell->n_children()==GeometryInfo<dim>::max_children_per_cell,
247  ExcNotImplemented());
248  for (unsigned int child=0; child<cell->n_children(); ++child)
249  {
250  // set an alias to the prolongation matrix for this child
251  prolongation
252  = mg_dof.get_fe().get_prolongation_matrix (child,
253  cell->refinement_case());
254 
255  if (this->mg_constrained_dofs != nullptr &&
256  this->mg_constrained_dofs->have_boundary_indices())
257  for (unsigned int j=0; j<dofs_per_cell; ++j)
258  if (this->mg_constrained_dofs->is_boundary_index(level, dof_indices_parent[j]))
259  for (unsigned int i=0; i<dofs_per_cell; ++i)
260  prolongation(i,j) = 0.;
261 
262  cell->child(child)->get_mg_dof_indices (dof_indices_child);
263 
264  // now set the entries in the matrix
265  for (unsigned int i=0; i<dofs_per_cell; ++i)
266  prolongation_matrices[level]->set (dof_indices_child[i],
267  dofs_per_cell,
268  &dof_indices_parent[0],
269  &prolongation(i,0),
270  true);
271  }
272  }
273  prolongation_matrices[level]->compress(VectorOperation::insert);
274  }
275 
276  this->fill_and_communicate_copy_indices(mg_dof);
277 }

295 template <typename VectorType>
296 std::size_t
297 MGTransferPrebuilt<VectorType>::memory_consumption () const
298 {
299  std::size_t result = MGLevelGlobalTransfer<VectorType>::memory_consumption();
300  for (unsigned int i=0; i<prolongation_matrices.size(); ++i)
301  result += prolongation_matrices[i]->memory_consumption()
302  + prolongation_sparsities[i]->memory_consumption();
303 
304  return result;
