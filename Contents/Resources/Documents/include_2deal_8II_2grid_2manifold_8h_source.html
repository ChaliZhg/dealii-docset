1195  template <typename MeshIteratorType>
1196  std::pair<std::vector<Point<MeshIteratorType::AccessorType::space_dimension> >,
1197  std::vector<double> >
1198  get_default_points_and_weights(const MeshIteratorType &iterator,
1199  const bool with_laplace)
1200  {
1201  const int spacedim = MeshIteratorType::AccessorType::space_dimension;
1202  const int dim = MeshIteratorType::AccessorType::structure_dimension;
1203 
1204  std::pair<std::vector<Point<spacedim> >,
1205  std::vector<double> > points_weights;
1206 
1207 
1208  // note that the exact weights are chosen such as to minimize the
1209  // distortion of the four new quads from the optimal shape; their
1210  // derivation and values is copied over from the
1211  // @p{MappingQ::set_laplace_on_vector} function
1212  switch (dim)
1213  {
1214  case 1:
1215  points_weights.first.resize(2);
1216  points_weights.second.resize(2);
1217  points_weights.first[0] = iterator->vertex(0);
1218  points_weights.second[0] = .5;
1219  points_weights.first[1] = iterator->vertex(1);
1220  points_weights.second[1] = .5;
1221  break;
1222  case 2:
1223  points_weights.first.resize(8);
1224  points_weights.second.resize(8);
1225 
1226  for (unsigned int i=0; i<4; ++i)
1227  {
1228  points_weights.first[i] = iterator->vertex(i);
1229  points_weights.first[4+i] = ( iterator->line(i)->has_children() ?
1230  iterator->line(i)->child(0)->vertex(1) :
1231  iterator->line(i)->get_manifold().get_new_point_on_line(iterator->line(i)) );
1232  }
1233 
1234  if (with_laplace)
1235  {
1236  std::fill(points_weights.second.begin(), points_weights.second.begin()+4, 1.0/16.0);
1237  std::fill(points_weights.second.begin()+4, points_weights.second.end(), 3.0/16.0);
1238  }
1239  else
1240  std::fill(points_weights.second.begin(), points_weights.second.end(), 1.0/8.0);
1241  break;
1242  case 3:
1243  {
1244  TriaIterator<TriaAccessor<3, 3, 3> > hex
1245  = static_cast<TriaIterator<TriaAccessor<3, 3, 3> > >(iterator);
1246  const unsigned int np =
1247  GeometryInfo<dim>::vertices_per_cell+
1248  GeometryInfo<dim>::lines_per_cell+
1249  GeometryInfo<dim>::faces_per_cell;
1250  points_weights.first.resize(np);
1251  points_weights.second.resize(np);
1252  std::vector<Point<3> > *sp3 = reinterpret_cast<std::vector<Point<3> > *>(&points_weights.first);
1253 
1254  unsigned int j=0;
1255 
1256  // note that the exact weights are chosen such as to minimize the
1257  // distortion of the eight new hexes from the optimal shape; their
1258  // derivation and values is copied over from the
1259  // @p{MappingQ::set_laplace_on_vector} function
1260  for (unsigned int i=0; i<GeometryInfo<dim>::vertices_per_cell; ++i, ++j)
1261  {
1262  (*sp3)[j] = hex->vertex(i);
1263  points_weights.second[j] = 1.0/128.0;
1264  }
1265  for (unsigned int i=0; i<GeometryInfo<dim>::lines_per_cell; ++i, ++j)
1266  {
1267  (*sp3)[j] = (hex->line(i)->has_children() ?
1268  hex->line(i)->child(0)->vertex(1) :
1269  hex->line(i)->get_manifold().get_new_point_on_line(hex->line(i)));
1270  points_weights.second[j] = 7.0/192.0;
1271  }
1272  for (unsigned int i=0; i<GeometryInfo<dim>::faces_per_cell; ++i, ++j)
1273  {
1274  (*sp3)[j] = (hex->quad(i)->has_children() ?
1275  hex->quad(i)->isotropic_child(0)->vertex(3) :
1276  hex->quad(i)->get_manifold().get_new_point_on_quad(hex->quad(i)));
1277  points_weights.second[j] = 1.0/12.0;
1278  }
1279  // Overwrite the weights with 1/np if we don't want to use
1280  // laplace vectors.
1281  if (with_laplace == false)
1282  std::fill(points_weights.second.begin(), points_weights.second.end(), 1.0/np);
1283  }
1284  break;
1285  default:
1286  Assert(false, ExcInternalError());
1287  break;
1288  }
1289  return points_weights;
1290  }
