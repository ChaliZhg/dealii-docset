346  struct CompareDownstream
347  {
351  CompareDownstream (const Tensor<1,dim> &dir)
352  :
353  dir(dir)
354  {}
358  bool operator () (const Iterator &c1, const Iterator &c2) const
359  {
360  const Tensor<1,dim> diff = c2->center() - c1->center();
361  return (diff*dir > 0);
362  }
363 
364  private:
368  const Tensor<1,dim> dir;
369  };

379  template <int dim>
380  struct ComparePointwiseDownstream
381  {
385  ComparePointwiseDownstream (const Tensor<1,dim> &dir)
386  :
387  dir(dir)
388  {}
392  bool operator () (const std::pair<Point<dim>,types::global_dof_index> &c1,
393  const std::pair<Point<dim>,types::global_dof_index> &c2) const
394  {
395  const Tensor<1,dim> diff = c2.first-c1.first;
396  return (diff*dir > 0 || (diff*dir==0 && c1.second<c2.second));
397  }
398 
399  private:
403  const Tensor<1,dim> dir;
404  };
