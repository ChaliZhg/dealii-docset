304  template <typename T>
305  class AlignedVectorMove : private parallel::ParallelForInteger
306  {
307  static const std::size_t minimum_parallel_grain_size = 160000/sizeof(T)+1;
308  public:
319  AlignedVectorMove (T *source_begin,
320  T *source_end,
321  T *destination,
322  const bool copy_source)
323  :
324  source_ (source_begin),
325  destination_ (destination),
326  copy_source_ (copy_source)
327  {
328  Assert (source_end >= source_begin, ExcInternalError());
329  const std::size_t size = source_end - source_begin;
330  if (size < minimum_parallel_grain_size)
331  apply_to_subrange (0, size);
332  else
333  apply_parallel (0, size, minimum_parallel_grain_size);
334  }
335 
340  virtual void apply_to_subrange (const std::size_t begin,
341  const std::size_t end) const
342  {
343  if (end == begin)
344  return;
345 
346  // for classes trivial assignment can use memcpy. cast element to
347  // (void*) to silence compiler warning for virtual classes (they will
348  // never arrive here because they are non-trivial).
349 
350  if (std::is_trivial<T>::value == true)
351  std::memcpy ((void *)(destination_+begin), (void *)(source_+begin),
352  (end-begin)*sizeof(T));
353  else if (copy_source_ == false)
354  for (std::size_t i=begin; i<end; ++i)
355  {
356  // initialize memory (copy construct by placement new), and
357  // destruct the source
358  new (&destination_[i]) T(source_[i]);
359  source_[i].~T();
360  }
361  else
362  for (std::size_t i=begin; i<end; ++i)
363  new (&destination_[i]) T(source_[i]);
364  }
365 
366  private:
367  T *source_;
368  T *destination_;
369  const bool copy_source_;
370  };

383  template <typename T, bool initialize_memory>
384  class AlignedVectorSet : private parallel::ParallelForInteger
385  {
386  static const std::size_t minimum_parallel_grain_size = 160000/sizeof(T)+1;
387  public:
392  AlignedVectorSet (const std::size_t size,
393  const T &element,
394  T *destination)
395  :
396  element_ (element),
397  destination_ (destination),
398  trivial_element (false)
399  {
400  if (size == 0)
401  return;
402 
403  // do not use memcmp for long double because on some systems it does not
404  // completely fill its memory and may lead to false positives in
405  // e.g. valgrind
406  if (std::is_trivial<T>::value == true &&
407  types_are_equal<T,long double>::value == false)
408  {
409  const unsigned char zero [sizeof(T)] = {};
410  // cast element to (void*) to silence compiler warning for virtual
411  // classes (they will never arrive here because they are
412  // non-trivial).
413  if (std::memcmp(zero, (void *)&element, sizeof(T)) == 0)
414  trivial_element = true;
415  }
416  if (size < minimum_parallel_grain_size)
417  apply_to_subrange (0, size);
418  else
419  apply_parallel (0, size, minimum_parallel_grain_size);
420  }
421 
425  virtual void apply_to_subrange (const std::size_t begin,
426  const std::size_t end) const
427  {
428  // for classes with trivial assignment of zero can use memset. cast
429  // element to (void*) to silence compiler warning for virtual
430  // classes (they will never arrive here because they are
431  // non-trivial).
432  if (std::is_trivial<T>::value == true && trivial_element)
433  std::memset ((void *)(destination_+begin), 0, (end-begin)*sizeof(T));
434  else
435  copy_construct_or_assign(begin, end,
436  ::internal::bool2type<initialize_memory>());
437  }
438 
439  private:
440  const T &element_;
441  mutable T *destination_;
442  bool trivial_element;
443 
444  // copy assignment operation
445  void copy_construct_or_assign(const std::size_t begin,
446  const std::size_t end,
447  ::internal::bool2type<false>) const
448  {
449  for (std::size_t i=begin; i<end; ++i)
450  destination_[i] = element_;
451  }
452 
453  // copy constructor (memory initialization)
454  void copy_construct_or_assign(const std::size_t begin,
455  const std::size_t end,
456  ::internal::bool2type<true>) const
457  {
458  for (std::size_t i=begin; i<end; ++i)
459  new (&destination_[i]) T(element_);
460  }
461  };

568 template < class T >
569 inline
570 void
571 AlignedVector<T>::resize_fast (const size_type size_in)
572 {
573  const size_type old_size = size();
574  if (std::is_trivial<T>::value == false && size_in < old_size)
575  {
576  // call destructor on fields that are released. doing it backward
577  // releases the elements in reverse order as compared to how they were
578  // created
579  while (_end_data != _data+size_in)
580  (--_end_data)->~T();
581  }
582  reserve (size_in);
583  _end_data = _data + size_in;
584 
585  // need to still set the values in case the class is non-trivial because
586  // virtual classes etc. need to run their (default) constructor
587  if (std::is_trivial<T>::value == false && size_in > old_size)
588  ::internal::AlignedVectorSet<T,true> (size_in-old_size, T(), _data+old_size);
589 }

596 AlignedVector<T>::resize (const size_type size_in,
597  const T &init)
598 {
599  const size_type old_size = size();
600  if (std::is_trivial<T>::value == false && size_in < old_size)

603  // releases the elements in reverse order as compared to how they were
604  // created

607  }

618 template < class T >
619 inline
620 void
621 AlignedVector<T>::reserve (const size_type size_alloc)
622 {
623  const size_type old_size = _end_data - _data;
624  const size_type allocated_size = _end_allocated - _data;
625  if (size_alloc > allocated_size)
626  {
627  // if we continuously increase the size of the vector, we might be
628  // reallocating a lot of times. therefore, try to increase the size more
629  // aggressively
630  size_type new_size = size_alloc;
631  if (size_alloc < (2 * allocated_size))
632  new_size = 2 * allocated_size;
633 
634  const size_type size_actual_allocate = new_size * sizeof(T);
635 
636  // allocate and align along 64-byte boundaries (this is enough for all
637  // levels of vectorization currently supported by deal.II)
638  T *new_data;
639  Utilities::System::posix_memalign ((void **)&new_data, 64, size_actual_allocate);
640 
641  // copy data in case there was some content before and release the old
642  // memory with the function corresponding to the one used for allocating
643  std::swap (_data, new_data);
644  _end_data = _data + old_size;
645  _end_allocated = _data + new_size;
646  if (_end_data != _data)
647  {
648  ::internal::AlignedVectorMove<T>(new_data, new_data + old_size,
649  _data, false);
650  free(new_data);
651  }
652  else
653  Assert(new_data == nullptr, ExcInternalError());
654  }
655  else if (size_alloc == 0)
656  clear();
657 }
