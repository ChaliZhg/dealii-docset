1134 inline
1135 void IndexSet::ElementIterator::advance ()
1136 {
1137  Assert(is_valid(), ExcMessage("Impossible to advance an IndexSet::ElementIterator that is invalid"));
1138  if (idx < index_set->ranges[range_idx].end)
1139  ++idx;
1140  // end of this range?
1141  if (idx == index_set->ranges[range_idx].end)
1142  {
1143  // point to first element in next interval if possible
1144  if (range_idx < index_set->ranges.size()-1)
1145  {
1146  ++range_idx;
1147  idx = index_set->ranges[range_idx].begin;
1148  }
1149  else
1150  {
1151  // we just fell off the end, set to invalid:
1152  range_idx = numbers::invalid_dof_index;
1153  idx = numbers::invalid_dof_index;
1154  }
1155  }
1156 }

1188 inline
1189 std::ptrdiff_t IndexSet::ElementIterator::operator - (const IndexSet::ElementIterator &other) const
1190 {
1191  Assert (index_set == other.index_set, ExcMessage("Can not compare iterators belonging to different IndexSets"));
1192  if (*this == other)
1193  return 0;
1194  if (!(*this < other))
1195  return -(other-*this);
1196 
1197  // only other can be equal to end() because of the checks above.
1198  Assert (is_valid(), ExcInternalError());
1199 
1200  // Note: we now compute how far advance *this in "*this < other" to get other, so we need to return -c at the end.
1201 
1202  // first finish the current range:
1203  std::ptrdiff_t c = index_set->ranges[range_idx].end-idx;
1204 
1205  // now walk in steps of ranges (need to start one behind our current one):
1206  for (size_type range=range_idx+1; range<index_set->ranges.size() && range<=other.range_idx; ++range)
1207  c += index_set->ranges[range].end-index_set->ranges[range].begin;
1208 
1209  Assert(other.range_idx < index_set->ranges.size() || other.range_idx == numbers::invalid_dof_index,
1210  ExcMessage("Inconsistent iterator state. Did you invalidate iterators by modifying the IndexSet?"));
1211 
1212  // We might have walked too far because we went until the end of other.range_idx, so walk backwards to other.idx:
1213  if (other.range_idx != numbers::invalid_dof_index)
1214  c -= index_set->ranges[other.range_idx].end - other.idx;
1215 
1216  return -c;
1217 }

1253 inline
1254 IndexSet::ElementIterator IndexSet::at(const size_type global_index) const
1255 {
1256  compress();
1257  Assert (global_index < size(),
1258  ExcIndexRangeType<size_type> (global_index, 0, size()));
1259 
1260  if (ranges.empty())
1261  return end();
1262 
1263  std::vector<Range>::const_iterator main_range=ranges.begin()+largest_range;
1264 
1265  Range r (global_index, global_index+1);
1266  // This optimization makes the bounds for lower_bound smaller by checking
1267  // the largest range first.
1268  std::vector<Range>::const_iterator range_begin, range_end;
1269  if (global_index<main_range->begin)
1270  {
1271  range_begin = ranges.begin();
1272  range_end = main_range;
1273  }
1274  else
1275  {
1276  range_begin = main_range;
1277  range_end = ranges.end();
1278  }
1279 
1280  // This will give us the first range p=[a,b[ with b>=global_index using
1281  // a binary search
1282  const std::vector<Range>::const_iterator
1283  p = Utilities::lower_bound(range_begin, range_end, r, Range::end_compare);
1284 
1285  // We couldn't find a range, which means we have no range that contains
1286  // global_index and also no range behind it, meaning we need to return end().
1287  if (p == ranges.end())
1288  return end();
1289 
1290  // Finally, we can have two cases: Either global_index is not in [a,b[,
1291  // which means we need to return an iterator to a because global_index, ...,
1292  // a-1 is not in the IndexSet (if branch). Alternatively, global_index is in
1293  // [a,b[ and we will return an iterator pointing directly at global_index
1294  // (else branch).
1295  if (global_index < p->begin)
1296  return IndexSet::ElementIterator(this, p-ranges.begin(), p->begin);
1297  else
1298  return IndexSet::ElementIterator(this, p-ranges.begin(), global_index);
1299 }

1483 inline
1484 bool
1485 IndexSet::is_element (const size_type index) const
1486 {
1487  if (ranges.empty() == false)
1488  {
1489  compress ();
1490 
1491  // fast check whether the index is in the largest range
1492  Assert (largest_range < ranges.size(), ExcInternalError());
1493  if (index >= ranges[largest_range].begin &&
1494  index < ranges[largest_range].end)
1495  return true;
1496 
1497  // get the element after which we would have to insert a range that
1498  // consists of all elements from this element to the end of the index
1499  // range plus one. after this call we know that if p!=end() then
1500  // p->begin<=index unless there is no such range at all
1501  //
1502  // if the searched for element is an element of this range, then we're
1503  // done. otherwise, the element can't be in one of the following ranges
1504  // because otherwise p would be a different iterator
1505  //
1506  // since we already know the position relative to the largest range (we
1507  // called compress!), we can perform the binary search on ranges with
1508  // lower/higher number compared to the largest range
1509  std::vector<Range>::const_iterator
1510  p = std::upper_bound (ranges.begin() + (index<ranges[largest_range].begin?
1511  0 : largest_range+1),
1512  index<ranges[largest_range].begin ?
1513  ranges.begin() + largest_range:
1514  ranges.end(),
1515  Range (index, size()+1));
1516 
1517  if (p == ranges.begin())
1518  return ((index >= p->begin) && (index < p->end));
1519 
1520  Assert ((p == ranges.end()) || (p->begin > index),
1521  ExcInternalError());
1522 
1523  // now move to that previous range
1524  --p;
1525  Assert (p->begin <= index, ExcInternalError());
1526 
1527  return (p->end > index);
1528  }
1529 
1530  // didn't find this index, so it's not in the set
1531  return false;
1532 }
