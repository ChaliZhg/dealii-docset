917  template <int dim, int spacedim>
918  template <typename number>
919  types::global_dof_index
920  DoFHandler<dim,spacedim>::n_boundary_dofs (const std::map<types::boundary_id, const Function<spacedim,number>*> &boundary_ids) const
921  {
922  // extract the set of boundary ids and forget about the function object pointers
923  std::set<types::boundary_id> boundary_ids_only;
924  for (typename std::map<types::boundary_id, const Function<spacedim,number>*>::const_iterator
925  p = boundary_ids.begin();
926  p != boundary_ids.end(); ++p)
927  boundary_ids_only.insert (p->first);
928 
929  // then just hand everything over to the other function that does the work
930  return n_boundary_dofs(boundary_ids_only);
931  }

953  template <int dim, int spacedim>
954  template <class Archive>
955  void DoFHandler<dim, spacedim>::load(Archive &ar, unsigned int)
956  {
957  ar &vertex_dofs;
958  ar &vertex_dofs_offsets;
959  ar &number_cache;
960 
961  // boost::serialization can restore pointers just fine, but if the
962  // pointer object still points to something useful, that object is not
963  // destroyed and we end up with a memory leak. consequently, first delete
964  // previous content before re-loading stuff
965  for (unsigned int i = 0; i<levels.size(); ++i)
966  delete levels[i];
967  for (unsigned int i = 0; i<has_children.size(); ++i)
968  delete has_children[i];
969  levels.resize(0);
970  has_children.resize(0);
971  delete faces;
972  faces = 0;
973 
974  ar &levels;
975  ar &faces;
976  ar &has_children;
977 
978  // these are the checks that correspond to the last block in the save()
979  // function
980  unsigned int n_cells;
981 
982  ar &n_cells;
983 
984  AssertThrow(n_cells == tria->n_cells(),
985  ExcMessage("The object being loaded into does not match the triangulation "
986  "that has been stored previously."));
987  }
