984 template <typename Accessor>
985 inline
986 bool
987 TriaRawIterator<Accessor>::operator < (const TriaRawIterator<Accessor> &other) const
988 {
989  Assert (state() != IteratorState::invalid, ExcDereferenceInvalidObject(accessor));
990  Assert (other.state() != IteratorState::invalid, ExcDereferenceInvalidObject(other.accessor));
991 
992  Assert (&accessor.get_triangulation() == &other.accessor.get_triangulation(),
993  ExcInvalidComparison());
994 
995  // Deal with iterators past end
996  if (state()==IteratorState::past_the_end)
997  return false;
998  if (other.state()==IteratorState::past_the_end)
999  return true;
1000 
1001  return ((**this) < (*other));
1002 }
