1447 inline
1448 bool
1449 SparsityPattern::operator == (const SparsityPattern &sp2) const
1450 {
1451  // it isn't quite necessary to compare *all* member variables. by only
1452  // comparing the essential ones, we can say that two sparsity patterns are
1453  // equal even if one is compressed and the other is not (in which case some
1454  // of the member variables are not yet set correctly)
1455  if (rows != sp2.rows ||
1456  cols != sp2.cols ||
1457  compressed != sp2.compressed ||
1458  store_diagonal_first_in_row != sp2.store_diagonal_first_in_row)
1459  return false;
1460 
1461  for (size_type i = 0; i < rows+1; ++i)
1462  if (rowstart[i] != sp2.rowstart[i])
1463  return false;
1464 
1465  for (size_type i = 0; i < rowstart[rows]; ++i)
1466  if (colnums[i] != sp2.colnums[i])
1467  return false;
1468 
1469  return true;
1470 }

1514 template <typename ForwardIterator>
1515 void
1516 SparsityPattern::copy_from (const size_type n_rows,
1517  const size_type n_cols,
1518  const ForwardIterator begin,
1519  const ForwardIterator end)
1520 {
1521  Assert (static_cast<size_type>(std::distance (begin, end)) == n_rows,
1522  ExcIteratorRange (std::distance (begin, end), n_rows));
1523 
1524  // first determine row lengths for each row. if the matrix is quadratic,
1525  // then we might have to add an additional entry for the diagonal, if that
1526  // is not yet present. as we have to call compress anyway later on, don't
1527  // bother to check whether that diagonal entry is in a certain row or not
1528  const bool is_square = (n_rows == n_cols);
1529  std::vector<unsigned int> row_lengths;
1530  row_lengths.reserve(n_rows);
1531  for (ForwardIterator i=begin; i!=end; ++i)
1532  row_lengths.push_back (std::distance (i->begin(), i->end())
1533  +
1534  (is_square ? 1 : 0));
1535  reinit (n_rows, n_cols, row_lengths);
1536 
1537  // now enter all the elements into the matrix. note that if the matrix is
1538  // quadratic, then we already have the diagonal element preallocated
1539  //
1540  // for use in the inner loop, we define a typedef to the type of the inner
1541  // iterators
1542  size_type row = 0;
1543  typedef typename std::iterator_traits<ForwardIterator>::value_type::const_iterator inner_iterator;
1544  for (ForwardIterator i=begin; i!=end; ++i, ++row)
1545  {
1546  size_type *cols = &colnums[rowstart[row]] + (is_square ? 1 : 0);
1547  const inner_iterator end_of_row = i->end();
1548  for (inner_iterator j=i->begin(); j!=end_of_row; ++j)
1549  {
1550  const size_type col
1551  = internal::SparsityPatternTools::get_column_index_from_iterator(*j);
1552  Assert (col < n_cols, ExcIndexRange(col,0,n_cols));
1553 
1554  if ((col!=row) || !is_square)
1555  *cols++ = col;
1556  }
1557  }
1558 
1559  // finally compress everything. this also sorts the entries within each row
1560  compress ();
1561 }
