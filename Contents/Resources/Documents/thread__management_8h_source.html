628  template <typename ForwardIterator>
629  std::vector<std::pair<ForwardIterator,ForwardIterator> >
630  split_range (const ForwardIterator &begin,
631  const ForwardIterator &end,
632  const unsigned int n_intervals)
633  {
634  typedef std::pair<ForwardIterator,ForwardIterator> IteratorPair;
635 
636  // in non-multithreaded mode, we often have the case that this
637  // function is called with n_intervals==1, so have a shortcut here
638  // to handle that case efficiently
639 
640  if (n_intervals==1)
641  return (std::vector<IteratorPair>
642  (1, IteratorPair(begin, end)));
643 
644  // if more than one interval requested, do the full work
645  const unsigned int n_elements = std::distance (begin, end);
646  const unsigned int n_elements_per_interval = n_elements / n_intervals;
647  const unsigned int residual = n_elements % n_intervals;
648 
649  std::vector<IteratorPair> return_values (n_intervals);
650 
651  return_values[0].first = begin;
652  for (unsigned int i=0; i<n_intervals; ++i)
653  {
654  if (i != n_intervals-1)
655  {
656  return_values[i].second = return_values[i].first;
657  // note: the cast is performed to avoid a warning of gcc
658  // that in the library `dist>=0' is checked (dist has a
659  // template type, which here is unsigned if no cast is
660  // performed)
661  std::advance (return_values[i].second,
662  static_cast<signed int>(n_elements_per_interval));
663  // distribute residual in division equally among the first
664  // few subintervals
665  if (i < residual)
666  ++return_values[i].second;
667 
668  return_values[i+1].first = return_values[i].second;
669  }
670  else
671  return_values[i].second = end;
672  }
673  return return_values;
674  }
675 }
