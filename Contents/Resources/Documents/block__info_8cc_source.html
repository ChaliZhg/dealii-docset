28 BlockInfo::initialize(const DoFHandler<dim, spacedim> &dof, bool levels_only, bool active_only)
29 {

37 
38  if (!active_only && dof.has_level_dofs())

44 
45  MGTools::count_dofs_per_block (dof, sizes);
46  levels.resize (sizes.size ());
47
