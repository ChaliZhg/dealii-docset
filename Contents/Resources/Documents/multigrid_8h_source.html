594 PreconditionMG<dim, VectorType, TRANSFER>::get_mpi_communicator() const
595 {
596  // currently parallel GMG works with distributed Triangulation only,
597  // so it should be a safe bet to use it to query MPI communicator:
598  const Triangulation<dim> &tria = dof_handler->get_triangulation();
599  const parallel::distributed::Triangulation<dim> *ptria = dynamic_cast<const parallel::distributed::Triangulation<dim> *>(&tria);
600  Assert (ptria != nullptr, ExcInternalError());
601  return ptria->get_communicator ();
602 }
