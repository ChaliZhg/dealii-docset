236 Number double_contract (const Tensor<2, dim, Number> &src1,
237  const Tensor<2, dim, Number> &src2)
238 {
239  Number res = 0.;
240  for (unsigned int i=0; i<dim; ++i)
241  res += src1[i] * src2[i];
242

246 template <int dim, typename Number>
247 inline
248 void double_contract (Tensor<2,dim,Number> &dest,
249  const Tensor<4,dim,Number> &src1,
250  const Tensor<2,dim,Number> &src2)
251 {
252  dest.clear ();
253  for (unsigned int i=0; i<dim; ++i)
254  for (unsigned int j=0; j<dim; ++j)
255  for (unsigned int k=0; k<dim; ++k)
256  for (unsigned int l=0; l<dim; ++l)
257  dest[i][j] += src1[i][j][k][l] * src2[k][l];
258 }
