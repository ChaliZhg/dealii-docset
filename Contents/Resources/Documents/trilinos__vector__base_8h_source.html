1275  inline
1276  void
1277  VectorBase::set (const size_type n_elements,
1278  const size_type *indices,
1279  const TrilinosScalar *values)
1280  {
1281  // if we have ghost values, do not allow
1282  // writing to this vector at all.
1283  Assert (!has_ghost_elements(), ExcGhostsPresent());
1284 
1285  if (last_action == Add)
1286  {
1287  const int ierr = vector->GlobalAssemble(Add);
1288  AssertThrow (ierr == 0, ExcTrilinosError(ierr));
1289  }
1290 
1291  if (last_action != Insert)
1292  last_action = Insert;
1293 
1294  for (size_type i=0; i<n_elements; ++i)
1295  {
1296  const size_type row = indices[i];
1297  const TrilinosWrappers::types::int_type local_row = vector->Map().LID(static_cast<TrilinosWrappers::types::int_type>(row));
1298  if (local_row != -1)
1299  (*vector)[0][local_row] = values[i];
1300  else
1301  {
1302  const int ierr = vector->ReplaceGlobalValues (1,
1303  (const TrilinosWrappers::types::int_type *)(&row),
1304  &values[i]);
1305  AssertThrow (ierr == 0, ExcTrilinosError(ierr));
1306  compressed = false;
1307  }
1308  // in set operation, do not use the pre-allocated vector for nonlocal
1309  // entries even if it exists. This is to ensure that we really only
1310  // set the elements touched by the set() method and not all contained
1311  // in the nonlocal entries vector (there is no way to distinguish them
1312  // on the receiving processor)
1313  }
1314  }

1350  inline
1351  void
1352  VectorBase::add (const size_type n_elements,
1353  const size_type *indices,
1354  const TrilinosScalar *values)
1355  {
1356  // if we have ghost values, do not allow
1357  // writing to this vector at all.
1358  Assert (!has_ghost_elements(), ExcGhostsPresent());
1359 
1360  if (last_action != Add)
1361  {
1362  if (last_action == Insert)
1363  {
1364  const int ierr = vector->GlobalAssemble(Insert);
1365  AssertThrow (ierr == 0, ExcTrilinosError(ierr));
1366  }
1367  last_action = Add;
1368  }
1369 
1370  for (size_type i=0; i<n_elements; ++i)
1371  {
1372  const size_type row = indices[i];
1373  const TrilinosWrappers::types::int_type local_row = vector->Map().LID(static_cast<TrilinosWrappers::types::int_type>(row));
1374  if (local_row != -1)
1375  (*vector)[0][local_row] += values[i];
1376  else if (nonlocal_vector.get() == nullptr)
1377  {
1378  const int ierr = vector->SumIntoGlobalValues (1,
1379  (const TrilinosWrappers::types::int_type *)(&row),
1380  &values[i]);
1381  AssertThrow (ierr == 0, ExcTrilinosError(ierr));
1382  compressed = false;
1383  }
1384  else
1385  {
1386  // use pre-allocated vector for non-local entries if it exists for
1387  // addition operation
1388  const TrilinosWrappers::types::int_type my_row = nonlocal_vector->Map().LID(static_cast<TrilinosWrappers::types::int_type>(row));
1389  Assert(my_row != -1,
1390  ExcMessage("Attempted to write into off-processor vector entry "
1391  "that has not be specified as being writable upon "
1392  "initialization"));
1393  (*nonlocal_vector)[0][my_row] += values[i];
1394  compressed = false;
1395  }
1396  }
1397  }
