171  AccessorBase::visit_present_row ()
172  {
173  // if we are asked to visit the past-the-end line, then simply
174  // release all our caches and go on with life.
175  //
176  // do the same if the row we're supposed to visit is not locally
177  // owned. this is simply going to make non-locally owned rows
178  // look like they're empty
179  if ((this->a_row == matrix->m())
180  ||
181  (matrix->in_local_range (this->a_row) == false))
182  {
183  colnum_cache.reset ();
184  value_cache.reset ();
185 
186  return;
187  }
188 
189  // get a representation of the present row
190  int ncols;
191  TrilinosWrappers::types::int_type colnums = matrix->n();
192  if (value_cache.get() == nullptr)
193  {
194  value_cache.reset (new std::vector<TrilinosScalar> (matrix->n()));
195  colnum_cache.reset (new std::vector<size_type> (matrix->n()));
196  }
197  else
198  {
199  value_cache->resize (matrix->n());
200  colnum_cache->resize (matrix->n());
201  }
202 
203  int ierr = matrix->trilinos_matrix().
204  ExtractGlobalRowCopy((TrilinosWrappers::types::int_type)this->a_row,
205  colnums,
206  ncols, &((*value_cache)[0]),
207  reinterpret_cast<TrilinosWrappers::types::int_type *>(&((*colnum_cache)[0])));
208  value_cache->resize (ncols);
209  colnum_cache->resize (ncols);
210  AssertThrow (ierr == 0, ExcTrilinosError(ierr));
211 
212  // copy it into our caches if the
213  // line isn't empty. if it is, then
214  // we've done something wrong, since
215  // we shouldn't have initialized an
216  // iterator for an empty line (what
217  // would it point to?)
218  }
219  }

434  void
435  SparseMatrix::copy_from (const SparseMatrix &rhs)
436  {
437  if (this == &rhs)
438  return;
439 
440  nonlocal_matrix.reset();
441  nonlocal_matrix_exporter.reset();
442 
443  // check whether we need to update the whole matrix layout (we have
444  // different maps or if we detect a row where the columns of the two
445  // matrices do not match)
446  bool needs_deep_copy =
447  !matrix->RowMap().SameAs(rhs.matrix->RowMap()) ||
448  !matrix->ColMap().SameAs(rhs.matrix->ColMap()) ||
449  !matrix->DomainMap().SameAs(rhs.matrix->DomainMap()) ||
450  n_nonzero_elements() != rhs.n_nonzero_elements();
451  if (!needs_deep_copy)
452  {
453  const std::pair<size_type, size_type>
454  local_range = rhs.local_range();
455 
456  // Try to copy all the rows of the matrix one by one. In case of error
457  // (i.e., the column indices are different), we need to abort and blow
458  // away the matrix.
459  for (size_type row=local_range.first; row < local_range.second; ++row)
460  {
461  const int row_local =
462  matrix->RowMap().LID(static_cast<TrilinosWrappers::types::int_type>(row));
463 
464  int n_entries, rhs_n_entries;
465  TrilinosScalar *value_ptr, *rhs_value_ptr;
466  int *index_ptr, *rhs_index_ptr;
467  int ierr = rhs.matrix->ExtractMyRowView (row_local, rhs_n_entries,
468  rhs_value_ptr, rhs_index_ptr);
469  (void)ierr;
470  Assert (ierr == 0, ExcTrilinosError(ierr));
471 
472  ierr = matrix->ExtractMyRowView (row_local, n_entries, value_ptr,
473  index_ptr);
474  Assert (ierr == 0, ExcTrilinosError(ierr));
475 
476  if (n_entries != rhs_n_entries ||
477  std::memcmp(static_cast<void *>(index_ptr),
478  static_cast<void *>(rhs_index_ptr),
479  sizeof(int)*n_entries) != 0)
480  {
481  needs_deep_copy = true;
482  break;
483  }
484 
485  for (int i=0; i<n_entries; ++i)
486  value_ptr[i] = rhs_value_ptr[i];
487  }
488  }
489 
490  if (needs_deep_copy)
491  {
492  column_space_map.reset (new Epetra_Map (rhs.domain_partitioner()));
493 
494  // release memory before reallocation
495  matrix.reset ();
496  matrix.reset (new Epetra_FECrsMatrix(*rhs.matrix));
497 
498  matrix->FillComplete(*column_space_map, matrix->RowMap());
499  }
500 
501  if (rhs.nonlocal_matrix.get() != nullptr)
502  nonlocal_matrix.reset(new Epetra_CrsMatrix(Copy, rhs.nonlocal_matrix->Graph()));
503  }

511  template <typename SparsityPatternType>
512  void
513  reinit_matrix (const Epetra_Map &input_row_map,
514  const Epetra_Map &input_col_map,
515  const SparsityPatternType &sparsity_pattern,
516  const bool exchange_data,
517  std::shared_ptr<Epetra_Map> &column_space_map,
518  std::shared_ptr<Epetra_FECrsMatrix> &matrix,
519  std::shared_ptr<Epetra_CrsMatrix> &nonlocal_matrix,
520  std::shared_ptr<Epetra_Export> &nonlocal_matrix_exporter)
521  {
522  // release memory before reallocation
523  matrix.reset();
524  nonlocal_matrix.reset();
525  nonlocal_matrix_exporter.reset();
526 
527  if (input_row_map.Comm().MyPID() == 0)
528  {
529  AssertDimension (sparsity_pattern.n_rows(),
530  static_cast<size_type>(n_global_elements(input_row_map)));
531  AssertDimension (sparsity_pattern.n_cols(),
532  static_cast<size_type>(n_global_elements(input_col_map)));
533  }
534 
535  column_space_map.reset (new Epetra_Map (input_col_map));
536 
537  // if we want to exchange data, build a usual Trilinos sparsity pattern
538  // and let that handle the exchange. otherwise, manually create a
539  // CrsGraph, which consumes considerably less memory because it can set
540  // correct number of indices right from the start
541  if (exchange_data)
542  {
543  SparsityPattern trilinos_sparsity;
544  trilinos_sparsity.reinit (input_row_map, input_col_map,
545  sparsity_pattern, exchange_data);
546  matrix.reset (new Epetra_FECrsMatrix
547  (Copy, trilinos_sparsity.trilinos_sparsity_pattern(), false));
548 
549  return;
550  }
551 
552  const size_type first_row = min_my_gid(input_row_map),
553  last_row = max_my_gid(input_row_map)+1;
554  std::vector<int> n_entries_per_row(last_row-first_row);
555 
556  for (size_type row=first_row; row<last_row; ++row)
557  n_entries_per_row[row-first_row] = sparsity_pattern.row_length(row);
558 
559  // The deal.II notation of a Sparsity pattern corresponds to the Epetra
560  // concept of a Graph. Hence, we generate a graph by copying the
561  // sparsity pattern into it, and then build up the matrix from the
562  // graph. This is considerable faster than directly filling elements
563  // into the matrix. Moreover, it consumes less memory, since the
564  // internal reordering is done on ints only, and we can leave the
565  // doubles aside.
566 
567  // for more than one processor, need to specify only row map first and
568  // let the matrix entries decide about the column map (which says which
569  // columns are present in the matrix, not to be confused with the
570  // col_map that tells how the domain dofs of the matrix will be
571  // distributed). for only one processor, we can directly assign the
572  // columns as well. Compare this with bug # 4123 in the Sandia Bugzilla.
573  std::shared_ptr<Epetra_CrsGraph> graph;
574  if (input_row_map.Comm().NumProc() > 1)
575  graph.reset (new Epetra_CrsGraph (Copy, input_row_map,
576  &n_entries_per_row[0], true));
577  else
578  graph.reset (new Epetra_CrsGraph (Copy, input_row_map, input_col_map,
579  &n_entries_per_row[0], true));
580 
581  // This functions assumes that the sparsity pattern sits on all
582  // processors (completely). The parallel version uses an Epetra graph
583  // that is already distributed.
584 
585  // now insert the indices
586  std::vector<TrilinosWrappers::types::int_type> row_indices;
587 
588  for (size_type row=first_row; row<last_row; ++row)
589  {
590  const int row_length = sparsity_pattern.row_length(row);
591  if (row_length == 0)
592  continue;
593 
594  row_indices.resize (row_length, -1);
595  {
596  typename SparsityPatternType::iterator p = sparsity_pattern.begin(row);
597  for (size_type col=0; p != sparsity_pattern.end(row); ++p, ++col)
598  row_indices[col] = p->column();
599  }
600  graph->Epetra_CrsGraph::InsertGlobalIndices (row, row_length,
601  &row_indices[0]);
602  }
603 
604  // Eventually, optimize the graph structure (sort indices, make memory
605  // contiguous, etc). note that the documentation of the function indeed
606  // states that we first need to provide the column (domain) map and then
607  // the row (range) map
608  graph->FillComplete(input_col_map, input_row_map);
609  graph->OptimizeStorage();
610 
611  // check whether we got the number of columns right.
612  AssertDimension (sparsity_pattern.n_cols(),
613  static_cast<size_type>(n_global_cols(*graph)));
614  (void)n_global_cols;
615 
616  // And now finally generate the matrix.
617  matrix.reset (new Epetra_FECrsMatrix(Copy, *graph, false));
618  }

664  AssertDimension (sparsity_pattern.n_cols(),
665  static_cast<size_type>(n_global_elements(input_col_map)));
666 
667  column_space_map.reset (new Epetra_Map (input_col_map));
668 
669  IndexSet relevant_rows (sparsity_pattern.row_index_set());
670  // serial case
671  if (relevant_rows.size() == 0)
672  {
673  relevant_rows.set_size(n_global_elements(input_row_map));
674  relevant_rows.add_range(0, n_global_elements(input_row_map));
675  }
676  relevant_rows.compress();
677  Assert(relevant_rows.n_elements() >= static_cast<unsigned int>(input_row_map.NumMyElements()),
678  ExcMessage("Locally relevant rows of sparsity pattern must contain "
679  "all locally owned rows"));
680 
681  // check whether the relevant rows correspond to exactly the same map as
682  // the owned rows. In that case, do not create the nonlocal graph and
683  // fill the columns by demand
684  bool have_ghost_rows = false;
685  {
686  std::vector<::types::global_dof_index> indices;
687  relevant_rows.fill_index_vector(indices);
688  Epetra_Map relevant_map (TrilinosWrappers::types::int_type(-1),
689  TrilinosWrappers::types::int_type(relevant_rows.n_elements()),
690  (indices.empty() ? nullptr :
691  reinterpret_cast<TrilinosWrappers::types::int_type *>(&indices[0])),
692  0, input_row_map.Comm());
693  if (relevant_map.SameAs(input_row_map))
694  have_ghost_rows = false;
695  else
696  have_ghost_rows = true;
697  }
698 
699  const unsigned int n_rows = relevant_rows.n_elements();
700  std::vector<TrilinosWrappers::types::int_type> ghost_rows;
701  std::vector<int> n_entries_per_row(input_row_map.NumMyElements());
702  std::vector<int> n_entries_per_ghost_row;
703  for (unsigned int i=0, own=0; i<n_rows; ++i)
704  {
705  const TrilinosWrappers::types::int_type global_row =
706  relevant_rows.nth_index_in_set(i);
707  if (input_row_map.MyGID(global_row))
708  n_entries_per_row[own++] = sparsity_pattern.row_length(global_row);
709  else if (sparsity_pattern.row_length(global_row) > 0)
710  {
711  ghost_rows.push_back(global_row);
712  n_entries_per_ghost_row.push_back(sparsity_pattern.row_length(global_row));
713  }
714  }
715 
716  Epetra_Map off_processor_map(-1, ghost_rows.size(),
717  (ghost_rows.size()>0)?(&ghost_rows[0]):nullptr,
718  0, input_row_map.Comm());
719 
720  std::shared_ptr<Epetra_CrsGraph> graph;
721  std::shared_ptr<Epetra_CrsGraphMod> nonlocal_graph;
722  if (input_row_map.Comm().NumProc() > 1)
723  {
724  graph.reset (new Epetra_CrsGraph (Copy, input_row_map,
725  (n_entries_per_row.size()>0)?(&n_entries_per_row[0]):nullptr,
726  exchange_data ? false : true));
727  if (have_ghost_rows == true)
728  nonlocal_graph.reset (new Epetra_CrsGraphMod (off_processor_map,
729  &n_entries_per_ghost_row[0]));
730  }
731  else
732  graph.reset (new Epetra_CrsGraph (Copy, input_row_map, input_col_map,
733  (n_entries_per_row.size()>0)?(&n_entries_per_row[0]):nullptr,
734  true));
735 
736  // now insert the indices, select between the right matrix
737  std::vector<TrilinosWrappers::types::int_type> row_indices;
738 
739  for (unsigned int i=0; i<n_rows; ++i)
740  {
741  const TrilinosWrappers::types::int_type global_row =
742  relevant_rows.nth_index_in_set(i);
743  const int row_length = sparsity_pattern.row_length(global_row);
744  if (row_length == 0)
745  continue;
746 
747  row_indices.resize (row_length, -1);
748  for (int col=0; col < row_length; ++col)
749  row_indices[col] = sparsity_pattern.column_number(global_row, col);
750 
751  if (input_row_map.MyGID(global_row))
752  graph->InsertGlobalIndices (global_row, row_length, &row_indices[0]);
753  else
754  {
755  Assert(nonlocal_graph.get() != nullptr, ExcInternalError());
756  nonlocal_graph->InsertGlobalIndices (global_row, row_length,
757  &row_indices[0]);
758  }
759  }
760 
761  // finalize nonlocal graph and create nonlocal matrix
762  if (nonlocal_graph.get() != nullptr)
763  {
764  // must make sure the IndicesAreGlobal flag is set on all processors
765  // because some processors might not call InsertGlobalIndices (and
766  // we do not want to insert dummy indices on all processors for
767  // large-scale simulations due to the bad impact on performance)
768  nonlocal_graph->SetIndicesAreGlobal();
769  Assert(nonlocal_graph->IndicesAreGlobal() == true,
770  ExcInternalError());
771  nonlocal_graph->FillComplete(input_col_map, input_row_map);
772  nonlocal_graph->OptimizeStorage();
773 
774  // insert data from nonlocal graph into the final sparsity pattern
775  if (exchange_data)
776  {
777  Epetra_Export exporter(nonlocal_graph->RowMap(), input_row_map);
778  int ierr = graph->Export(*nonlocal_graph, exporter, Add);
779  (void)ierr;
780  Assert (ierr==0, ExcTrilinosError(ierr));
781  }
782 
783  nonlocal_matrix.reset (new Epetra_CrsMatrix(Copy, *nonlocal_graph));
784  }
785 
786  graph->FillComplete(input_col_map, input_row_map);
787  graph->OptimizeStorage();
788 
789  AssertDimension (sparsity_pattern.n_cols(),static_cast<size_type>(
790  n_global_cols(*graph)));
791 
792  matrix.reset (new Epetra_FECrsMatrix(Copy, *graph, false));
793  }

924  inline
925  void SparseMatrix::reinit (const IndexSet &row_parallel_partitioning,
926  const IndexSet &col_parallel_partitioning,
927  const ::SparseMatrix<number> &dealii_sparse_matrix,
928  const MPI_Comm &communicator,
929  const double drop_tolerance,
930  const bool copy_values,
931  const ::SparsityPattern *use_this_sparsity)
932  {
933  if (copy_values == false)
934  {
935  // in case we do not copy values, just
936  // call the other function.
937  if (use_this_sparsity == nullptr)
938  reinit (row_parallel_partitioning, col_parallel_partitioning,
939  dealii_sparse_matrix.get_sparsity_pattern(),
940  communicator, false);
941  else
942  reinit (row_parallel_partitioning, col_parallel_partitioning,
943  *use_this_sparsity, communicator, false);
944  return;
945  }
946 
947  const size_type n_rows = dealii_sparse_matrix.m();
948 
949  AssertDimension (row_parallel_partitioning.size(), n_rows);
950  AssertDimension (col_parallel_partitioning.size(), dealii_sparse_matrix.n());
951 
952  const ::SparsityPattern &sparsity_pattern =
953  (use_this_sparsity!=nullptr)? *use_this_sparsity :
954  dealii_sparse_matrix.get_sparsity_pattern();
955 
956  if (matrix.get() == nullptr ||
957  m() != n_rows ||
958  n_nonzero_elements() != sparsity_pattern.n_nonzero_elements())
959  {
960  reinit (row_parallel_partitioning, col_parallel_partitioning,
961  sparsity_pattern, communicator, false);
962  }
963 
964  // fill the values. the same as above: go through all rows of the
965  // matrix, and then all columns. since the sparsity patterns of the
966  // input matrix and the specified sparsity pattern might be different,
967  // need to go through the row for both these sparsity structures
968  // simultaneously in order to really set the correct values.
969  size_type maximum_row_length = matrix->MaxNumEntries();
970  std::vector<size_type> row_indices (maximum_row_length);
971  std::vector<TrilinosScalar> values (maximum_row_length);
972 
973  for (size_type row=0; row<n_rows; ++row)
974  // see if the row is locally stored on this processor
975  if (row_parallel_partitioning.is_element(row) == true)
976  {
977  ::SparsityPattern::iterator select_index =
978  sparsity_pattern.begin(row);
979  typename ::SparseMatrix<number>::const_iterator it =
980  dealii_sparse_matrix.begin(row);
981  size_type col = 0;
982  if (sparsity_pattern.n_rows() == sparsity_pattern.n_cols())
983  {
984  // optimized diagonal
985  AssertDimension(it->column(), row);
986  if (std::fabs(it->value()) > drop_tolerance)
987  {
988  values[col] = it->value();
989  row_indices[col++] = it->column();
990  }
991  ++select_index;
992  ++it;
993  }
994 
995  while (it != dealii_sparse_matrix.end(row) &&
996  select_index != sparsity_pattern.end(row))
997  {
998  while (select_index->column() < it->column() &&
999  select_index != sparsity_pattern.end(row))
1000  ++select_index;
1001  while (it->column() < select_index->column() &&
1002  it != dealii_sparse_matrix.end(row))
1003  ++it;
1004 
1005  if (it == dealii_sparse_matrix.end(row))
1006  break;
1007  if (std::fabs(it->value()) > drop_tolerance)
1008  {
1009  values[col] = it->value();
1010  row_indices[col++] = it->column();
1011  }
1012  ++select_index;
1013  ++it;
1014  }
1015  set (row, col, reinterpret_cast<size_type *>(&row_indices[0]),
1016  &values[0], false);
1017  }
1018  compress(VectorOperation::insert);
1019  }

1221 
1222  TrilinosScalar
1223  SparseMatrix::operator() (const size_type i,
1224  const size_type j) const
1225  {
1226  // Extract local indices in
1227  // the matrix.
1228  int trilinos_i = matrix->LRID(static_cast<TrilinosWrappers::types::int_type>(i)),
1229  trilinos_j = matrix->LCID(static_cast<TrilinosWrappers::types::int_type>(j));
1230  TrilinosScalar value = 0.;
1231 
1232  // If the data is not on the
1233  // present processor, we throw
1234  // an exception. This is one of
1235  // the two tiny differences to
1236  // the el(i,j) call, which does
1237  // not throw any assertions.
1238  if (trilinos_i == -1)
1239  {
1240  Assert (false, ExcAccessToNonLocalElement(i, j, local_range().first,
1241  local_range().second));
1242  }
1243  else
1244  {
1245  // Check whether the matrix has
1246  // already been transformed to local
1247  // indices.
1248  Assert (matrix->Filled(), ExcMatrixNotCompressed());
1249 
1250  // Prepare pointers for extraction
1251  // of a view of the row.
1252  int nnz_present = matrix->NumMyEntries(trilinos_i);
1253  int nnz_extracted;
1254  int *col_indices;
1255  TrilinosScalar *values;
1256 
1257  // Generate the view and make
1258  // sure that we have not generated
1259  // an error.
1260  // TODO Check that col_indices are int and not long long
1261  int ierr = matrix->ExtractMyRowView(trilinos_i, nnz_extracted,
1262  values, col_indices);
1263  (void)ierr;
1264  Assert (ierr==0, ExcTrilinosError(ierr));
1265 
1266  Assert (nnz_present == nnz_extracted,
1267  ExcDimensionMismatch(nnz_present, nnz_extracted));
1268 
1269  // Search the index where we
1270  // look for the value, and then
1271  // finally get it.
1272 
1273  int *el_find = std::find(col_indices, col_indices + nnz_present, trilinos_j);
1274 
1275  int local_col_index = (int)(el_find - col_indices);
1276 
1277  // This is actually the only
1278  // difference to the el(i,j)
1279  // function, which means that
1280  // we throw an exception in
1281  // this case instead of just
1282  // returning zero for an
1283  // element that is not present
1284  // in the sparsity pattern.
1285  if (local_col_index == nnz_present)
1286  {
1287  Assert (false, ExcInvalidIndex (i,j));
1288  }
1289  else
1290  value = values[local_col_index];
1291  }
1292
