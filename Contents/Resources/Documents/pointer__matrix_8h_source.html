769 inline void
770 PointerMatrixAux<MatrixType, VectorType>::vmult_add (VectorType &dst,
771  const VectorType &src) const
772 {
773  if (mem == nullptr)
774  mem = &my_memory;
775  Assert (mem != nullptr, ExcNotInitialized());
776  Assert (m != nullptr, ExcNotInitialized());
777  VectorType *v = mem->alloc();
