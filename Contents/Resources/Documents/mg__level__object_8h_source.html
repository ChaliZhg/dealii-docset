190 template<class Object>
191 void
192 MGLevelObject<Object>::resize (const unsigned int new_minlevel,
193  const unsigned int new_maxlevel)
194 {
195  Assert (new_minlevel <= new_maxlevel, ExcInternalError());
196  // note that on clear(), the
197  // shared_ptr class takes care of
198  // deleting the object it points to
199  // by itself
200  objects.clear ();
201 
202  minlevel = new_minlevel;
203  for (unsigned int i=0; i<new_maxlevel-new_minlevel+1; ++i)
204  objects.push_back(std::shared_ptr<Object> (new Object));
205 }

259  typedef typename std::vector<std::shared_ptr<Object> >::const_iterator Iter;
260  const Iter end = objects.end();
