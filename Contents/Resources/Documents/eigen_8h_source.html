231 template <class VectorType>
232 template <typename MatrixType>
233 void
234 EigenPower<VectorType>::solve (double &value,
235  const MatrixType &A,
236  VectorType &x)
237 {
238  SolverControl::State conv=SolverControl::iterate;
239 
240  deallog.push("Power method");
241 
242  VectorType *Vy = this->memory.alloc ();
243  VectorType &y = *Vy;
244  y.reinit (x);
245  VectorType *Vr = this->memory.alloc ();
246  VectorType &r = *Vr;
247  r.reinit (x);
248 
249  double length = x.l2_norm ();
250  double old_length = 0.;
251  x *= 1./length;
252 
253  A.vmult (y,x);
254 
255  // Main loop
256  int iter=0;
257  for (; conv==SolverControl::iterate; iter++)
258  {
259  y.add(additional_data.shift, x);
260 
261  // Compute absolute value of eigenvalue
262  old_length = length;
263  length = y.l2_norm ();
264 
265  // do a little trick to compute the sign
266  // with not too much effect of round-off errors.
267  double entry = 0.;
268  size_type i = 0;
269  double thresh = length/x.size();
270  do
271  {
272  Assert (i<x.size(), ExcInternalError());
273  entry = y (i++);
274  }
275  while (std::fabs(entry) < thresh);
276 
277  --i;
278 
279  // Compute unshifted eigenvalue
280  value = (entry * x (i) < 0.) ? -length : length;
281  value -= additional_data.shift;
282 
283  // Update normalized eigenvector
284  x.equ (1/length, y);
285 
286  // Compute residual
287  A.vmult (y,x);
288 
289  // Check the change of the eigenvalue
290  // Brrr, this is not really a good criterion
291  conv = this->iteration_status (iter, std::fabs(1./length-1./old_length), x);
292  }
293 
294  this->memory.free(Vy);
295  this->memory.free(Vr);
296 
297  deallog.pop();
298 
299  // in case of failure: throw exception
300  AssertThrow(conv == SolverControl::success, SolverControl::NoConvergence (iter,
301  std::fabs(1./length-1./old_length)));
302 
303  // otherwise exit as normal
304 }

325 template <class VectorType>
326 template <typename MatrixType>
327 void
328 EigenInverse<VectorType>::solve (double &value,
329  const MatrixType &A,
330  VectorType &x)
331 {
332  deallog.push("Wielandt");
333 
334  SolverControl::State conv=SolverControl::iterate;
335 
336  // Prepare matrix for solver
337  auto A_op = linear_operator(A);
338  double current_shift = -value;
339  auto A_s = A_op + current_shift * identity_operator(A_op);
340 
341  // Define solver
342  ReductionControl inner_control (5000, 1.e-16, 1.e-5, false, false);
343  PreconditionIdentity prec;
344  SolverGMRES<VectorType>
345  solver(inner_control, this->memory);
346 
347  // Next step for recomputing the shift
348  unsigned int goal = additional_data.start_adaption;
349 
350  // Auxiliary vector
351  VectorType *Vy = this->memory.alloc ();
352  VectorType &y = *Vy;
353  y.reinit (x);
354  VectorType *Vr = this->memory.alloc ();
355  VectorType &r = *Vr;
356  r.reinit (x);
357 
358  double length = x.l2_norm ();
359  double old_value = value;
360 
361  x *= 1./length;
362 
363  // Main loop
364  double res = -std::numeric_limits<double>::max();
365  size_type iter=0;
366  for (; conv==SolverControl::iterate; iter++)
367  {
368  solver.solve (A_s, y, x, prec);
369 
370  // Compute absolute value of eigenvalue
371  length = y.l2_norm ();
372 
373  // do a little trick to compute the sign
374  // with not too much effect of round-off errors.
375  double entry = 0.;
376  size_type i = 0;
377  double thresh = length/x.size();
378  do
379  {
380  Assert (i<x.size(), ExcInternalError());
381  entry = y (i++);
382  }
383  while (std::fabs(entry) < thresh);
384 
385  --i;
386 
387  // Compute unshifted eigenvalue
388  value = (entry * x(i) < 0. ? -1. : 1.) / length - current_shift;
389 
390  if (iter==goal)
391  {
392  const auto &relaxation = additional_data.relaxation;
393  const double new_shift =
394  relaxation * (-value) + (1. - relaxation) * current_shift;
395 
396  A_s = A_op + new_shift * identity_operator(A_op);
397  current_shift = new_shift;
398 
399  ++goal;
400  }
401 
402  // Update normalized eigenvector
403  x.equ (1./length, y);
404  // Compute residual
405  if (additional_data.use_residual)
406  {
407  y.equ (value, x);
408  A.vmult(r,x);
409  r.sadd(-1., value, x);
410  res = r.l2_norm();
411  // Check the residual
412  conv = this->iteration_status (iter, res, x);
413  }
414  else
415  {
416  res = std::fabs(1./value-1./old_value);
417  conv = this->iteration_status (iter, res, x);
418  }
419  old_value = value;
420  }
421 
422  this->memory.free(Vy);
423  this->memory.free(Vr);
424 
425  deallog.pop();
426 
427  // in case of failure: throw
428  // exception
429  AssertThrow (conv == SolverControl::success,
430  SolverControl::NoConvergence (iter,
431  res));
432  // otherwise exit as normal
433 }
