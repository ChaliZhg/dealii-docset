1254 template <int dim, int spacedim>
1255 template <typename number>
1256 types::global_dof_index
1257 DoFHandler<dim,spacedim>::n_boundary_dofs (const std::map<types::boundary_id, const Function<spacedim,number>*> &boundary_ids) const
1258 {
1259  // extract the set of boundary ids and forget about the function object pointers
1260  std::set<types::boundary_id> boundary_ids_only;
1261  for (typename std::map<types::boundary_id, const Function<spacedim,number>*>::const_iterator
1262  p = boundary_ids.begin();
1263  p != boundary_ids.end(); ++p)
1264  boundary_ids_only.insert (p->first);
1265 
1266  // then just hand everything over to the other function that does the work
1267  return n_boundary_dofs(boundary_ids_only);
1268 }

1308 template <int dim, int spacedim>
1309 template <class Archive>
1310 void DoFHandler<dim,spacedim>::load (Archive &ar,
1311  const unsigned int)
1312 {
1313  ar &block_info_object;
1314  ar &vertex_dofs;
1315  ar &number_cache;
1316 
1317  // boost::serialization can restore pointers just fine, but if the
1318  // pointer object still points to something useful, that object is not
1319  // destroyed and we end up with a memory leak. consequently, first delete
1320  // previous content before re-loading stuff
1321  for (unsigned int i=0; i<levels.size(); ++i)
1322  delete levels[i];
1323  levels.resize (0);
1324  delete faces;
1325  faces = nullptr;
1326 
1327  ar &levels;
1328  ar &faces;
1329 
1330  // these are the checks that correspond to the last block in the save()
1331  // function
1332  unsigned int n_cells;
1333  std::string fe_name;
1334  std::string policy_name;
1335 
1336  ar &n_cells &fe_name &policy_name;
1337 
1338  AssertThrow (n_cells == tria->n_cells(),
1339  ExcMessage ("The object being loaded into does not match the triangulation "
1340  "that has been stored previously."));
1341  AssertThrow (fe_name == selected_fe->get_name(),
1342  ExcMessage ("The finite element associated with this DoFHandler does not match "
1343  "the one that was associated with the DoFHandler previously stored."));
1344  AssertThrow (policy_name == internal::policy_to_string(*policy),
1345  ExcMessage (std::string ("The policy currently associated with this DoFHandler (")
1346  + internal::policy_to_string(*policy)
1347  +std::string(") does not match the one that was associated with the "
1348  "DoFHandler previously stored (")
1349  + policy_name
1350  + ")."));
