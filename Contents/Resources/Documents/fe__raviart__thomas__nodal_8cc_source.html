35 template <int dim>
36 FE_RaviartThomasNodal<dim>::FE_RaviartThomasNodal (const unsigned int deg)
37  :
38  FE_PolyTensor<PolynomialsRaviartThomas<dim>, dim> (
39  deg,
40  FiniteElementData<dim>(get_dpo_vector(deg),
41  dim, deg+1, FiniteElementData<dim>::Hdiv),
42  get_ria_vector (deg),
43  std::vector<ComponentMask>(PolynomialsRaviartThomas<dim>::compute_n_pols(deg),
44  std::vector<bool>(dim,true)))
45 {
46  Assert (dim >= 2, ExcImpossibleInDim(dim));
47  const unsigned int n_dofs = this->dofs_per_cell;
48 
49  this->mapping_type = mapping_raviart_thomas;
50  // First, initialize the
51  // generalized support points and
52  // quadrature weights, since they
53  // are required for interpolation.
54  initialize_support_points(deg);
55 
56  // Now compute the inverse node matrix, generating the correct
57  // basis functions from the raw ones. For a discussion of what
58  // exactly happens here, see FETools::compute_node_matrix.
59  const FullMatrix<double> M = FETools::compute_node_matrix(*this);
60  this->inverse_node_matrix.reinit(n_dofs, n_dofs);
61  this->inverse_node_matrix.invert(M);
62  // From now on, the shape functions provided by FiniteElement::shape_value
63  // and similar functions will be the correct ones, not
64  // the raw shape functions from the polynomial space anymore.
65 
66  // Reinit the vectors of
67  // prolongation matrices to the
68  // right sizes. There are no
69  // restriction matrices implemented
70  for (unsigned int ref_case=RefinementCase<dim>::cut_x;
71  ref_case<RefinementCase<dim>::isotropic_refinement+1; ++ref_case)
72  {
73  const unsigned int nc = GeometryInfo<dim>::n_children(RefinementCase<dim>(ref_case));
74 
75  for (unsigned int i=0; i<nc; ++i)
76  this->prolongation[ref_case-1][i].reinit (n_dofs, n_dofs);
77  }
78  // Fill prolongation matrices with embedding operators
79  FETools::compute_embedding_matrices (*this, this->prolongation);
80  // TODO[TL]: for anisotropic refinement we will probably need a table of submatrices with an array for each refine case
81  FullMatrix<double> face_embeddings[GeometryInfo<dim>::max_children_per_face];
82  for (unsigned int i=0; i<GeometryInfo<dim>::max_children_per_face; ++i)
83  face_embeddings[i].reinit (this->dofs_per_face, this->dofs_per_face);
84  FETools::compute_face_embedding_matrices<dim,double>(*this, face_embeddings, 0, 0);
85  this->interface_constraints.reinit((1<<(dim-1)) * this->dofs_per_face,
86  this->dofs_per_face);
87  unsigned int target_row=0;
88  for (unsigned int d=0; d<GeometryInfo<dim>::max_children_per_face; ++d)
89  for (unsigned int i=0; i<face_embeddings[d].m(); ++i)
90  {
91  for (unsigned int j=0; j<face_embeddings[d].n(); ++j)
92  this->interface_constraints(target_row,j) = face_embeddings[d](i,j);
93  ++target_row;
94  }
95 }

135 template <int dim>
136 void
137 FE_RaviartThomasNodal<dim>::initialize_support_points (const unsigned int deg)
138 {
139  this->generalized_support_points.resize (this->dofs_per_cell);
140  this->generalized_face_support_points.resize (this->dofs_per_face);
141 
142  // Number of the point being entered
143  unsigned int current = 0;
144 
145  // On the faces, we choose as many
146  // Gauss points as necessary to
147  // determine the normal component
148  // uniquely. This is the deg of
149  // the Raviart-Thomas element plus
150  // one.
151  if (dim>1)
152  {
153  QGauss<dim-1> face_points (deg+1);
154  Assert (face_points.size() == this->dofs_per_face,
155  ExcInternalError());
156  for (unsigned int k=0; k<this->dofs_per_face; ++k)
157  this->generalized_face_support_points[k] = face_points.point(k);
158  Quadrature<dim> faces = QProjector<dim>::project_to_all_faces(face_points);
159  for (unsigned int k=0;
160  k<this->dofs_per_face*GeometryInfo<dim>::faces_per_cell;
161  ++k)
162  this->generalized_support_points[k] = faces.point(k+QProjector<dim>
163  ::DataSetDescriptor::face(0,
164  true,
165  false,
166  false,
167  this->dofs_per_face));
168 
169  current = this->dofs_per_face*GeometryInfo<dim>::faces_per_cell;
170  }
171 
172  if (deg==0) return;
173  // In the interior, we need
174  // anisotropic Gauss quadratures,
175  // different for each direction.
176  QGauss<1> high(deg+1);
177  QGauss<1> low(deg);
178 
179  for (unsigned int d=0; d<dim; ++d)
180  {
181  QAnisotropic<dim> *quadrature;
182  if (dim == 1) quadrature = new QAnisotropic<dim>(high);
183  if (dim == 2) quadrature = new QAnisotropic<dim>(((d==0) ? low : high),
184  ((d==1) ? low : high));
185  if (dim == 3) quadrature = new QAnisotropic<dim>(((d==0) ? low : high),
186  ((d==1) ? low : high),
187  ((d==2) ? low : high));
188  Assert(dim<=3, ExcNotImplemented());
189 
190  for (unsigned int k=0; k<quadrature->size(); ++k)
191  this->generalized_support_points[current++] = quadrature->point(k);
192  delete quadrature;
193  }
194  Assert (current == this->dofs_per_cell, ExcInternalError());
195 }

199 template <int dim>
200 std::vector<unsigned int>
201 FE_RaviartThomasNodal<dim>::get_dpo_vector (const unsigned int deg)
202 {
203  // the element is face-based and we have
204  // (deg+1)^(dim-1) DoFs per face
205  unsigned int dofs_per_face = 1;
206  for (unsigned int d=1; d<dim; ++d)
207  dofs_per_face *= deg+1;
208 
209  // and then there are interior dofs
210  const unsigned int
211  interior_dofs = dim*deg*dofs_per_face;
212 
213  std::vector<unsigned int> dpo(dim+1);
214  dpo[dim-1] = dofs_per_face;
215  dpo[dim] = interior_dofs;
216 
217  return dpo;
218 }

232 template <int dim>
233 std::vector<bool>
234 FE_RaviartThomasNodal<dim>::get_ria_vector (const unsigned int deg)
235 {
236  const unsigned int dofs_per_cell = PolynomialsRaviartThomas<dim>::compute_n_pols(deg);
237  unsigned int dofs_per_face = deg+1;
238  for (unsigned int d=2; d<dim; ++d)
239  dofs_per_face *= deg+1;
240  // all face dofs need to be
241  // non-additive, since they have
242  // continuity requirements.
243  // however, the interior dofs are
244  // made additive
245  std::vector<bool> ret_val(dofs_per_cell,false);
246  for (unsigned int i=GeometryInfo<dim>::faces_per_cell*dofs_per_face;
247  i < dofs_per_cell; ++i)
248  ret_val[i] = true;
249 
250  return ret_val;
251 }

254 template <int dim>
255 bool
256 FE_RaviartThomasNodal<dim>::has_support_on_face (
257  const unsigned int shape_index,
258  const unsigned int face_index) const
259 {
260  Assert (shape_index < this->dofs_per_cell,
261  ExcIndexRange (shape_index, 0, this->dofs_per_cell));
262  Assert (face_index < GeometryInfo<dim>::faces_per_cell,
263  ExcIndexRange (face_index, 0, GeometryInfo<dim>::faces_per_cell));
264 
265  // The first degrees of freedom are
266  // on the faces and each face has
267  // degree degrees.
268  const unsigned int support_face = shape_index / this->degree;
269 
270  // The only thing we know for sure
271  // is that shape functions with
272  // support on one face are zero on
273  // the opposite face.
274  if (support_face < GeometryInfo<dim>::faces_per_cell)
275  return (face_index != GeometryInfo<dim>::opposite_face[support_face]);
276 
277  // In all other cases, return true,
278  // which is safe
279  return true;
280 }

284 template <int dim>
285 void
286 FE_RaviartThomasNodal<dim>::
287 convert_generalized_support_point_values_to_nodal_values(const std::vector<Vector<double> > &support_point_values,
288  std::vector<double> &nodal_values) const
289 {
290  Assert (support_point_values.size() == this->generalized_support_points.size(),
291  ExcDimensionMismatch(support_point_values.size(), this->generalized_support_points.size()));
292  Assert (nodal_values.size() == this->dofs_per_cell,
293  ExcDimensionMismatch(nodal_values.size(),this->dofs_per_cell));
294  Assert (support_point_values[0].size() == this->n_components(),
295  ExcDimensionMismatch(support_point_values[0].size(),this->n_components()));
296 
297  // First do interpolation on
298  // faces. There, the component
299  // evaluated depends on the face
300  // direction and orientation.
301  unsigned int fbase = 0;
302  unsigned int f=0;
303  for (; f<GeometryInfo<dim>::faces_per_cell;
304  ++f, fbase+=this->dofs_per_face)
305  {
306  for (unsigned int i=0; i<this->dofs_per_face; ++i)
307  {
308  nodal_values[fbase+i] = support_point_values[fbase+i](GeometryInfo<dim>::unit_normal_direction[f]);
309  }
310  }
311 
312  // The remaining points form dim
313  // chunks, one for each component.
314  const unsigned int istep = (this->dofs_per_cell - fbase) / dim;
315  Assert ((this->dofs_per_cell - fbase) % dim == 0, ExcInternalError());
316 
317  f = 0;
318  while (fbase < this->dofs_per_cell)
319  {
320  for (unsigned int i=0; i<istep; ++i)
321  {
322  nodal_values[fbase+i] = support_point_values[fbase+i](f);
323  }
324  fbase+=istep;
325  ++f;
326  }
327  Assert (fbase == this->dofs_per_cell, ExcInternalError());
328 }

363 template <int dim>
364 std::vector<std::pair<unsigned int, unsigned int> >
365 FE_RaviartThomasNodal<dim>::
366 hp_line_dof_identities (const FiniteElement<dim> &fe_other) const
367 {
368  // we can presently only compute
369  // these identities if both FEs are
370  // FE_RaviartThomasNodals
371  if (const FE_RaviartThomasNodal<dim> *fe_q_other
372  = dynamic_cast<const FE_RaviartThomasNodal<dim>*>(&fe_other))
373  {
374  // dofs are located on faces; these are
375  // only lines in 2d
376  if (dim != 2)
377  return std::vector<std::pair<unsigned int, unsigned int> >();
378 
379  // dofs are located along lines, so two
380  // dofs are identical only if in the
381  // following two cases (remember that
382  // the face support points are Gauss
383  // points):
384  //1. this->degree = fe_q_other->degree,
385  // in the case, all the dofs on
386  // the line are identical
387  //2. this->degree-1 and fe_q_other->degree-1
388  // are both even, i.e. this->dof_per_line
389  // and fe_q_other->dof_per_line are both odd,
390  // there exists only one point (the middle one)
391  // such that dofs are identical on this point
392  //
393  // to understand this, note that
394  // this->degree is the *maximal*
395  // polynomial degree, and is thus one
396  // higher than the argument given to
397  // the constructor
398  const unsigned int p = this->degree-1;
399  const unsigned int q = fe_q_other->degree-1;
400 
401  std::vector<std::pair<unsigned int, unsigned int> > identities;
402 
403  if (p==q)
404  for (unsigned int i=0; i<p+1; ++i)
405  identities.emplace_back (i, i);
406 
407  else if (p%2==0 && q%2==0)
408  identities.emplace_back(p/2, q/2);
409 
410  return identities;
411  }
412  else
413  {
414  Assert (false, ExcNotImplemented());
415  return std::vector<std::pair<unsigned int, unsigned int> > ();
416  }
417 }

504 template <int dim>
505 void
506 FE_RaviartThomasNodal<dim>::get_face_interpolation_matrix (
507  const FiniteElement<dim> &x_source_fe,
508  FullMatrix<double> &interpolation_matrix) const
509 {
510  // this is only implemented, if the
511  // source FE is also a
512  // RaviartThomasNodal element
513  AssertThrow ((x_source_fe.get_name().find ("FE_RaviartThomasNodal<") == 0)
514  ||
515  (dynamic_cast<const FE_RaviartThomasNodal<dim>*>(&x_source_fe) != nullptr),
516  typename FiniteElement<dim>::
517  ExcInterpolationNotImplemented());
518 
519  Assert (interpolation_matrix.n() == this->dofs_per_face,
520  ExcDimensionMismatch (interpolation_matrix.n(),
521  this->dofs_per_face));
522  Assert (interpolation_matrix.m() == x_source_fe.dofs_per_face,
523  ExcDimensionMismatch (interpolation_matrix.m(),
524  x_source_fe.dofs_per_face));
525 
526  // ok, source is a RaviartThomasNodal element, so
527  // we will be able to do the work
528  const FE_RaviartThomasNodal<dim> &source_fe
529  = dynamic_cast<const FE_RaviartThomasNodal<dim>&>(x_source_fe);
530 
531  // Make sure, that the element,
532  // for which the DoFs should be
533  // constrained is the one with
534  // the higher polynomial degree.
535  // Actually the procedure will work
536  // also if this assertion is not
537  // satisfied. But the matrices
538  // produced in that case might
539  // lead to problems in the
540  // hp procedures, which use this
541  // method.
542  Assert (this->dofs_per_face <= source_fe.dofs_per_face,
543  typename FiniteElement<dim>::
544  ExcInterpolationNotImplemented ());
545 
546  // generate a quadrature
547  // with the generalized support points.
548  // This is later based as a
549  // basis for the QProjector,
550  // which returns the support
551  // points on the face.
552  Quadrature<dim-1> quad_face_support (source_fe.get_generalized_face_support_points ());
553 
554  // Rule of thumb for FP accuracy,
555  // that can be expected for a
556  // given polynomial degree.
557  // This value is used to cut
558  // off values close to zero.
559  double eps = 2e-13*this->degree*(dim-1);
560 
561  // compute the interpolation
562  // matrix by simply taking the
563  // value at the support points.
564  const Quadrature<dim> face_projection
565  = QProjector<dim>::project_to_face (quad_face_support, 0);
566 
567  for (unsigned int i=0; i<source_fe.dofs_per_face; ++i)
568  {
569  const Point<dim> &p = face_projection.point (i);
570 
571  for (unsigned int j=0; j<this->dofs_per_face; ++j)
572  {
573  double matrix_entry
574  = this->shape_value_component (this->face_to_cell_index(j, 0),
575  p, 0);
576 
577  // Correct the interpolated
578  // value. I.e. if it is close
579  // to 1 or 0, make it exactly
580  // 1 or 0. Unfortunately, this
581  // is required to avoid problems
582  // with higher order elements.
583  if ( std::fabs(matrix_entry - 1.0) < eps )
584  matrix_entry = 1.0;
585  if ( std::fabs(matrix_entry) < eps )
586  matrix_entry = 0.0;
587 
588  interpolation_matrix(i,j) = matrix_entry;
589  }
590  }
591 
592  // make sure that the row sum of
593  // each of the matrices is 1 at
594  // this point. this must be so
595  // since the shape functions sum up
596  // to 1
597  for (unsigned int j=0; j<source_fe.dofs_per_face; ++j)
598  {
599  double sum = 0.;
600 
601  for (unsigned int i=0; i<this->dofs_per_face; ++i)
602  sum += interpolation_matrix(j,i);
603 
604  Assert (std::fabs(sum-1) < 2e-13*this->degree*(dim-1),
605  ExcInternalError());
606  }
607 }

610 template <int dim>
611 void
612 FE_RaviartThomasNodal<dim>::get_subface_interpolation_matrix (
613  const FiniteElement<dim> &x_source_fe,
614  const unsigned int subface,
615  FullMatrix<double> &interpolation_matrix) const
616 {
617  // this is only implemented, if the
618  // source FE is also a
619  // RaviartThomasNodal element
620  AssertThrow ((x_source_fe.get_name().find ("FE_RaviartThomasNodal<") == 0)
621  ||
622  (dynamic_cast<const FE_RaviartThomasNodal<dim>*>(&x_source_fe) != nullptr),
623  typename FiniteElement<dim>::
624  ExcInterpolationNotImplemented());
625 
626  Assert (interpolation_matrix.n() == this->dofs_per_face,
627  ExcDimensionMismatch (interpolation_matrix.n(),
628  this->dofs_per_face));
629  Assert (interpolation_matrix.m() == x_source_fe.dofs_per_face,
630  ExcDimensionMismatch (interpolation_matrix.m(),
631  x_source_fe.dofs_per_face));
632 
633  // ok, source is a RaviartThomasNodal element, so
634  // we will be able to do the work
635  const FE_RaviartThomasNodal<dim> &source_fe
636  = dynamic_cast<const FE_RaviartThomasNodal<dim>&>(x_source_fe);
637 
638  // Make sure, that the element,
639  // for which the DoFs should be
640  // constrained is the one with
641  // the higher polynomial degree.
642  // Actually the procedure will work
643  // also if this assertion is not
644  // satisfied. But the matrices
645  // produced in that case might
646  // lead to problems in the
647  // hp procedures, which use this
648  // method.
649  Assert (this->dofs_per_face <= source_fe.dofs_per_face,
650  typename FiniteElement<dim>::
651  ExcInterpolationNotImplemented ());
652 
653  // generate a quadrature
654  // with the generalized support points.
655  // This is later based as a
656  // basis for the QProjector,
657  // which returns the support
658  // points on the face.
659  Quadrature<dim-1> quad_face_support (source_fe.get_generalized_face_support_points ());
660 
661  // Rule of thumb for FP accuracy,
662  // that can be expected for a
663  // given polynomial degree.
664  // This value is used to cut
665  // off values close to zero.
666  double eps = 2e-13*this->degree*(dim-1);
667 
668  // compute the interpolation
669  // matrix by simply taking the
670  // value at the support points.
671 
672  const Quadrature<dim> subface_projection
673  = QProjector<dim>::project_to_subface (quad_face_support, 0, subface);
674 
675  for (unsigned int i=0; i<source_fe.dofs_per_face; ++i)
676  {
677  const Point<dim> &p = subface_projection.point (i);
678 
679  for (unsigned int j=0; j<this->dofs_per_face; ++j)
680  {
681  double matrix_entry
682  = this->shape_value_component (this->face_to_cell_index(j, 0), p, 0);
683 
684  // Correct the interpolated
685  // value. I.e. if it is close
686  // to 1 or 0, make it exactly
687  // 1 or 0. Unfortunately, this
688  // is required to avoid problems
689  // with higher order elements.
690  if ( std::fabs(matrix_entry - 1.0) < eps )
691  matrix_entry = 1.0;
692  if ( std::fabs(matrix_entry) < eps )
693  matrix_entry = 0.0;
694 
695  interpolation_matrix(i,j) = matrix_entry;
696  }
697  }
698 
699  // make sure that the row sum of
700  // each of the matrices is 1 at
701  // this point. this must be so
702  // since the shape functions sum up
703  // to 1
704  for (unsigned int j=0; j<source_fe.dofs_per_face; ++j)
705  {
706  double sum = 0.;
707 
708  for (unsigned int i=0; i<this->dofs_per_face; ++i)
709  sum += interpolation_matrix(j,i);
710 
711  Assert (std::fabs(sum-1) < 2e-13*this->degree*(dim-1),
712  ExcInternalError());
713  }
