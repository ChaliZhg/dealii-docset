1243  ExcIteratorPastEnd());
1244

1248  // step, then cycle until we
1249  // find a row with a nonzero
1250  // number of entries.

1252  {

1257  &&

1427  template <typename ForwardIterator>
1428  inline
1429  void
1430  SparsityPattern::add_entries (const size_type row,
1431  ForwardIterator begin,
1432  ForwardIterator end,
1433  const bool /*indices_are_sorted*/)
1434  {
1435  if (begin == end)
1436  return;
1437 
1438  // verify that the size of the data type Trilinos expects matches that the
1439  // iterator points to. we allow for some slippage between signed and
1440  // unsigned and only compare that they are both either 32 or 64 bit. to
1441  // write this test properly, not that we cannot compare the size of
1442  // '*begin' because 'begin' may be an iterator and '*begin' may be an
1443  // accessor class. consequently, we need to somehow get an actual value
1444  // from it which we can by evaluating an expression such as when
1445  // multiplying the value produced by 2
1446  Assert (sizeof(TrilinosWrappers::types::int_type) ==
1447  sizeof((*begin)*2),
1448  ExcNotImplemented());
1449 
1450  TrilinosWrappers::types::int_type *col_index_ptr =
1451  (TrilinosWrappers::types::int_type *)(&*begin);
1452  const int n_cols = static_cast<int>(end - begin);
1453 
1454  int ierr;
1455  if ( graph->RowMap().LID(static_cast<TrilinosWrappers::types::int_type>(row)) != -1)
1456  ierr = graph->InsertGlobalIndices (row, n_cols, col_index_ptr);
1457  else if (nonlocal_graph.get() != nullptr)
1458  {
1459  // this is the case when we have explicitly set the off-processor rows
1460  // and want to create a separate matrix object for them (to retain
1461  // thread-safety)
1462  Assert (nonlocal_graph->RowMap().LID(static_cast<TrilinosWrappers::types::int_type>(row)) != -1,
1463  ExcMessage("Attempted to write into off-processor matrix row "
1464  "that has not be specified as being writable upon "
1465  "initialization"));
1466  ierr = nonlocal_graph->InsertGlobalIndices (row, n_cols, col_index_ptr);
1467  }
1468  else
1469  ierr = graph->InsertGlobalIndices
1470  (1, (TrilinosWrappers::types::int_type *)&row, n_cols, col_index_ptr);
1471 
1472  AssertThrow (ierr >= 0, ExcTrilinosError(ierr));
1473  }
