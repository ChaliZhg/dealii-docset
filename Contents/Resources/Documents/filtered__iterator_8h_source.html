780  // The following classes create a nested sequencee of
781  // FilteredIterator<FilteredIterator<...<BaseIterator>...>> with as many
782  // levels of FilteredIterator classes as there are elements in the TypeList
783  // if the latter is given as a std::tuple<Args...>.

785  struct NestFilteredIterators;
