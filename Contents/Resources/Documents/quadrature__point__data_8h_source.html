498 template <typename CellIteratorType, typename DataType>
499 template<typename T>
500 void CellDataStorage<CellIteratorType,DataType>::initialize(const CellIteratorType &cell,
501  const unsigned int n_q_points)
502 {
503  static_assert(std::is_base_of<DataType, T>::value,
504  "User's T class should be derived from user's DataType class");
505 
506  if (map.find(cell) == map.end())
507  {
508  map[cell] = std::vector<std::shared_ptr<DataType> >(n_q_points);
509  // we need to initialise one-by-one as the std::vector<>(q, T())
510  // will end with a single same T object stored in each element of the vector:
511  auto it = map.find(cell);
512  for (unsigned int q=0; q < n_q_points; q++)
513  it->second[q] = std::make_shared<T>();
514  }
515 }

548 template <typename CellIteratorType, typename DataType>
549 void CellDataStorage<CellIteratorType,DataType>::clear()
550 {
551  // Do not call
552  // map.clear();
553  // as we want to be sure no one uses the stored objects. Loop manually:
554  auto it = map.begin();
555  while (it != map.end())
556  {
557  // loop over all objects and see if noone is using them
558  for (unsigned int i = 0; i < it->second.size(); i++)
559  {
560  Assert(it->second[i].unique(),
561  ExcMessage("Can not erase the cell data, multiple objects reference it."));
562  }
563  it = map.erase(it);
564  }

570 template <typename CellIteratorType, typename DataType>
571 template <typename T>
572 std::vector<std::shared_ptr<T> >
573 CellDataStorage<CellIteratorType,DataType>::get_data(const CellIteratorType &cell)
574 {
575  static_assert(std::is_base_of<DataType, T>::value,
576  "User's T class should be derived from user's DataType class");
577 
578  auto it = map.find(cell);
579  Assert(it != map.end(), ExcMessage("Could not find data for the cell"));
580 
581  // It would be nice to have a specialized version of this function for T==DataType.
582  // However explicit (i.e full) specialization of a member template is only allowed
583  // when the enclosing class is also explicitly (i.e fully) specialized.
584  // Thus, stick with copying of shared pointers even when the T==DataType:
585  std::vector<std::shared_ptr<T>> res (it->second.size());
586  for (unsigned int q = 0; q < res.size(); q++)
587  res[q] = std::dynamic_pointer_cast<T>(it->second[q]);
588  return res;
589 }

594 template <typename CellIteratorType, typename DataType>
595 template <typename T>
596 std::vector<std::shared_ptr<const T> >
597 CellDataStorage<CellIteratorType,DataType>::get_data(const CellIteratorType &cell) const
598 {
599  static_assert(std::is_base_of<DataType, T>::value,
600  "User's T class should be derived from user's DataType class");
601 
602  auto it = map.find(cell);
603  Assert(it != map.end(), ExcMessage("Could not find QP data for the cell"));
604 
605  // Cast base class to the desired class. This has to be done irrespectively of
606  // T==DataType as we need to return shapred_ptr<const T> to make sure the user
607  // does not modify the content of QP objects
608  std::vector<std::shared_ptr<const T>> res (it->second.size());
609  for (unsigned int q = 0; q < res.size(); q++)
610  res[q] = std::dynamic_pointer_cast<const T>(it->second[q]);
611  return res;
612 }

619 /*
620  * Pack cell data of type @p DataType stored using @p data_storage in @p cell
621  * at each quadrature point to @p matrix_data. Here @p matrix_data is a matrix
622  * whose first index corresponds to different quadrature points on the cell
623  * whereas the second index represents different values stored at each
624  * quadrature point in the DataType class.
625  */
626 template<typename CellIteratorType, typename DataType>
627 void pack_cell_data
628 (const CellIteratorType &cell,
629  const CellDataStorage<CellIteratorType,DataType> *data_storage,
630  FullMatrix<double> &matrix_data)
631 {
632  static_assert(std::is_base_of<TransferableQuadraturePointData, DataType>::value,
633  "User's DataType class should be derived from QPData");
634 
635  const std::vector<std::shared_ptr<const DataType> > qpd = data_storage->get_data(cell);
636 
637  const unsigned int n = matrix_data.n();
638 
639  std::vector<double> single_qp_data(n);
640  Assert (qpd.size() == matrix_data.m(),
641  ExcDimensionMismatch(qpd.size(), matrix_data.m()));
642  for (unsigned int q = 0; q < qpd.size(); q++)
643  {
644  qpd[q]->pack_values(single_qp_data);
645  Assert (single_qp_data.size() == n,
646  ExcDimensionMismatch(single_qp_data.size(),n));
647 
648  for (unsigned int i = 0; i < n; i++)
649  matrix_data(q,i) = single_qp_data[i];
650  }
651 }

659 template<typename CellIteratorType, typename DataType>
660 void unpack_to_cell_data
661 (const CellIteratorType &cell,
662  const FullMatrix<double> &values_at_qp,
663  CellDataStorage<CellIteratorType,DataType> *data_storage)
664 {
665  static_assert(std::is_base_of<TransferableQuadraturePointData, DataType>::value,
666  "User's DataType class should be derived from QPData");
667 
668  std::vector<std::shared_ptr<DataType> > qpd = data_storage->get_data(cell);
669 
670  const unsigned int n = values_at_qp.n();
671 
672  std::vector<double> single_qp_data(n);
673  Assert(qpd.size() == values_at_qp.m(),
674  ExcDimensionMismatch(qpd.size(), values_at_qp.m()));
675 
676  for (unsigned int q = 0; q < qpd.size(); q++)
677  {
678  for (unsigned int i = 0; i < n; i++)
679  single_qp_data[i] = values_at_qp(q,i);
680  qpd[q]->unpack_values(single_qp_data);
681  }
682 }

729  template<int dim, typename DataType>
730  void ContinuousQuadratureDataTransfer<dim,DataType>::prepare_for_coarsening_and_refinement
731  (parallel::distributed::Triangulation<dim> &tr_,
732  CellDataStorage<CellIteratorType,DataType> &data_storage_)
733  {
734  Assert (data_storage == nullptr,
735  ExcMessage("This function can be called only once"));
736  triangulation = &tr_;
737  data_storage = &data_storage_;
738  // get the number from the first active cell
739  unsigned int number_of_values = 0;
740  // if triangulation has some active cells locally owned cells on this processor we can expect
741  // data to be initialized. Do that to get the number:
742  for (typename parallel::distributed::Triangulation<dim>::active_cell_iterator it = triangulation->begin_active();
743  it != triangulation->end(); it++)
744  if (it->is_locally_owned())
745  {
746  std::vector<std::shared_ptr<DataType> > qpd = data_storage->get_data(it);
747  number_of_values = qpd[0]->number_of_values();
748  break;
749  }
750  // some processors may have no data stored, thus get the maximum among all processors:
751  number_of_values = Utilities::MPI::max(number_of_values, triangulation->get_communicator ());
752  Assert (number_of_values > 0,
753  ExcInternalError());
754  const unsigned int dofs_per_cell = projection_fe->dofs_per_cell;
755  matrix_dofs.reinit(dofs_per_cell,number_of_values);
756  matrix_dofs_child.reinit(dofs_per_cell,number_of_values);
757  matrix_quadrature.reinit(n_q_points,number_of_values);
758  data_size_in_bytes = sizeof(double) * dofs_per_cell * number_of_values;
759 
760  offset = triangulation->register_data_attach(data_size_in_bytes,
761  std::bind(&ContinuousQuadratureDataTransfer<dim,DataType>::pack_function,
762  this,
763  std::placeholders::_1,
764  std::placeholders::_2,
765  std::placeholders::_3));
766  }

787  template<int dim, typename DataType>
788  void ContinuousQuadratureDataTransfer<dim,DataType>::pack_function
789  (const typename parallel::distributed::Triangulation<dim,dim>::cell_iterator &cell,
790  const typename parallel::distributed::Triangulation<dim,dim>::CellStatus /*status*/,
791  void *data)
792  {
793  double *data_store = reinterpret_cast<double *>(data);

795  pack_cell_data(cell,data_storage,matrix_quadrature);

805  template<int dim, typename DataType>
806  void ContinuousQuadratureDataTransfer<dim,DataType>::
807  unpack_function (const typename parallel::distributed::Triangulation<dim,dim>::cell_iterator &cell,
808  const typename parallel::distributed::Triangulation<dim,dim>::CellStatus status,
809  const void *data)
810  {
811  Assert ((status!=parallel::distributed::Triangulation<dim,dim>::CELL_COARSEN),
812  ExcNotImplemented());
813  (void) status;
814 
815  const double *data_store = reinterpret_cast<const double *>(data);
816 
817  std::memcpy(&matrix_dofs(0,0), data_store, data_size_in_bytes);
818 
819  if (cell->has_children())
820  {
821  // we need to first use prolongation matrix to get dofvalues on child
822  // cells based on dofvalues stored in the parent's data_store
823  for (unsigned int child=0; child<cell->n_children(); ++child)
824  if (cell->child(child)->is_locally_owned())
825  {
826  projection_fe->get_prolongation_matrix(child, cell->refinement_case())
827  .mmult (matrix_dofs_child, matrix_dofs);
828 
829  // now we do the usual business of evaluating FE on quadrature points:
830  project_to_qp_matrix.mmult(matrix_quadrature,matrix_dofs_child);
831 
832  // finally, put back into the map:
833  unpack_to_cell_data(cell->child(child),matrix_quadrature,data_storage);
834  }
835  }
836  else
837  {
838  // if there are no children, evaluate FE field at
839  // rhs_quadrature points.
840  project_to_qp_matrix.mmult(matrix_quadrature,matrix_dofs);
841 
842  // finally, put back into the map:
843  unpack_to_cell_data(cell,matrix_quadrature,data_storage);
844  }
845  }
