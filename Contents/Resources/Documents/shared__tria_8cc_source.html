57  template <int dim, int spacedim>
58  void Triangulation<dim,spacedim>::partition()
59  {
60  if (settings & partition_metis)
61  {
62  ::GridTools::partition_triangulation (this->n_subdomains, *this);
63  }
64  else if (settings & partition_zorder)
65  {
66  ::GridTools::partition_triangulation_zorder (this->n_subdomains, *this);
67  }
68  else if (settings & partition_custom_signal)
69  {
70  // User partitions mesh manually
71  }
72  else
73  {
74  AssertThrow(false, ExcInternalError())
75  }
76 
77  // do not partition multigrid levels if user is
78  // defining a custom partition
79  if ((settings & construct_multigrid_hierarchy) && !(settings & partition_custom_signal))
80  ::GridTools::partition_multigrid_levels(*this);
81 
82  true_subdomain_ids_of_cells.resize(this->n_active_cells());
83 
84  // loop over all cells and mark artificial:
85  typename parallel::shared::Triangulation<dim,spacedim>::active_cell_iterator
86  cell = this->begin_active(),
87  endc = this->end();
88 
89  if (allow_artificial_cells)
90  {
91  // get active halo layer of (ghost) cells
92  // parallel::shared::Triangulation<dim>::
93  std::function<bool (const typename parallel::shared::Triangulation<dim,spacedim>::active_cell_iterator &)> predicate
94  = IteratorFilters::SubdomainEqualTo(this->my_subdomain);
95 
96  const std::vector<typename parallel::shared::Triangulation<dim,spacedim>::active_cell_iterator>
97  active_halo_layer_vector = GridTools::compute_active_cell_halo_layer (*this, predicate);
98  std::set<typename parallel::shared::Triangulation<dim,spacedim>::active_cell_iterator>
99  active_halo_layer(active_halo_layer_vector.begin(), active_halo_layer_vector.end());
100 
101  for (unsigned int index=0; cell != endc; cell++, index++)
102  {
103  // store original/true subdomain ids:
104  true_subdomain_ids_of_cells[index] = cell->subdomain_id();
105 
106  if (cell->is_locally_owned() == false &&
107  active_halo_layer.find(cell) == active_halo_layer.end())
108  cell->set_subdomain_id(numbers::artificial_subdomain_id);
109  }
110 
111  // loop over all cells in multigrid hierarchy and mark artificial:
112  if (settings & construct_multigrid_hierarchy)
113  {
114  std::function<bool (const typename parallel::shared::Triangulation<dim,spacedim>::cell_iterator &)> predicate
115  = IteratorFilters::LocallyOwnedLevelCell();
116  for (unsigned int lvl=0; lvl<this->n_levels(); ++lvl)
117  {
118  const std::vector<typename parallel::shared::Triangulation<dim,spacedim>::cell_iterator>
119  level_halo_layer_vector = GridTools::compute_cell_halo_layer_on_level (*this, predicate, lvl);
120  std::set<typename parallel::shared::Triangulation<dim,spacedim>::cell_iterator>
121  level_halo_layer(level_halo_layer_vector.begin(), level_halo_layer_vector.end());
122 
123  typename parallel::shared::Triangulation<dim,spacedim>::cell_iterator
124  cell = this->begin(lvl),
125  endc = this->end(lvl);
126  for (; cell != endc; cell++)
127  {
128  // for active cells, we must have knowledge of level subdomain ids of
129  // all neighbors to our subdomain, not just neighbors on the same level.
130  // if the cells subdomain id was not set to artitficial above, we will
131  // also keep its level subdomain id since it is either owned by this processor
132  // or in the ghost layer of the active mesh.
133  if (!cell->has_children() && cell->subdomain_id() != numbers::artificial_subdomain_id)
134  continue;
135 
136  // we must have knowledge of our parent in the hierarchy
137  if (cell->has_children())
138  {
139  bool keep_cell = false;
140  for (unsigned int c=0; c<GeometryInfo<dim>::max_children_per_cell; ++c)
141  if (cell->child(c)->level_subdomain_id() == this->my_subdomain)
142  {
143  keep_cell = true;
144  break;
145  }
146  if (keep_cell)
147  continue;
148  }
149 
150  // we must have knowledge of our neighbors on the same level
151  if (!cell->is_locally_owned_on_level() &&
152  level_halo_layer.find(cell) != level_halo_layer.end())
153  continue;
154 
155  // mark all other cells to artificial
156  cell->set_level_subdomain_id(numbers::artificial_subdomain_id);
157  }
158  }
159  }
160  }
161  else
162  {
163  // just store true subdomain ids
164  for (unsigned int index=0; cell != endc; cell++, index++)
165  true_subdomain_ids_of_cells[index] = cell->subdomain_id();
166  }
167 
168 #ifdef DEBUG
169  {
170  // Assert that each cell is owned by a processor
171  unsigned int n_my_cells = 0;
172  typename parallel::shared::Triangulation<dim,spacedim>::active_cell_iterator
173  cell = this->begin_active(),
174  endc = this->end();
175  for (; cell!=endc; ++cell)
176  if (cell->is_locally_owned())
177  n_my_cells += 1;
178 
179  unsigned int total_cells;
180  int ierr = MPI_Allreduce (&n_my_cells, &total_cells, 1,
181  MPI_UNSIGNED, MPI_SUM, this->get_communicator());
182  AssertThrowMPI(ierr);
183 
184  Assert(total_cells == this->n_active_cells(),
185  ExcMessage("Not all cells are assigned to a processor."))
186  }
187 
188  // If running with multigrid, assert that each level
189  // cell is owned by a processor
190  if (settings & construct_multigrid_hierarchy)
191  {
192  unsigned int n_my_cells = 0;
193  typename parallel::shared::Triangulation<dim,spacedim>::cell_iterator
194  cell = this->begin(),
195  endc = this->end();
196  for (; cell!=endc; ++cell)
197  if (cell->is_locally_owned_on_level())
198  n_my_cells += 1;
199 
200  unsigned int total_cells;
201  int ierr = MPI_Allreduce (&n_my_cells, &total_cells, 1, MPI_UNSIGNED, MPI_SUM, this->get_communicator());
202  AssertThrowMPI(ierr);
203 
204  Assert(total_cells == this->n_cells(),
205  ExcMessage("Not all cells are assigned to a processor."))
206  }
207 #endif
208  }
