1185 template <int dim, typename Number>
1186 inline
1187 std::pair<unsigned int,unsigned int>
1188 MatrixFree<dim,Number>::create_cell_subrange_hp
1189 (const std::pair<unsigned int,unsigned int> &range,
1190  const unsigned int degree,
1191  const unsigned int vector_component) const
1192 {
1193  AssertIndexRange (vector_component, dof_info.size());
1194  if (dof_info[vector_component].cell_active_fe_index.empty())
1195  {
1196  AssertDimension (dof_info[vector_component].fe_index_conversion.size(),1);
1197  if (dof_info[vector_component].fe_index_conversion[0].first == degree)
1198  return range;
1199  else
1200  return std::pair<unsigned int,unsigned int> (range.second,range.second);
1201  }
1202 
1203  const unsigned int fe_index =
1204  dof_info[vector_component].fe_index_from_degree(degree);
1205  if (fe_index >= dof_info[vector_component].max_fe_index)
1206  return std::pair<unsigned int,unsigned int>(range.second, range.second);
1207  else
1208  return create_cell_subrange_hp_by_index (range, fe_index, vector_component);
1209 }

1213 template <int dim, typename Number>
1214 inline
1215 std::pair<unsigned int,unsigned int>
1216 MatrixFree<dim,Number>::create_cell_subrange_hp_by_index
1217 (const std::pair<unsigned int,unsigned int> &range,
1218  const unsigned int fe_index,
1219  const unsigned int vector_component) const
1220 {
1221  AssertIndexRange (fe_index, dof_info[vector_component].max_fe_index);
1222  const std::vector<unsigned int> &fe_indices =
1223  dof_info[vector_component].cell_active_fe_index;
1224  if (fe_indices.size() == 0)
1225  return range;
1226  else
1227  {
1228  // the range over which we are searching must be ordered, otherwise we
1229  // got a range that spans over too many cells
1230 #ifdef DEBUG
1231  for (unsigned int i=range.first+1; i<range.second; ++i)
1232  Assert (fe_indices[i] >= fe_indices[i-1],
1233  ExcMessage ("Cell range must be over sorted range of fe indices in hp case!"));
1234  AssertIndexRange(range.first,fe_indices.size()+1);
1235  AssertIndexRange(range.second,fe_indices.size()+1);
1236 #endif
1237  std::pair<unsigned int,unsigned int> return_range;
1238  return_range.first =
1239  std::lower_bound(&fe_indices[0] + range.first,
1240  &fe_indices[0] + range.second, fe_index)
1241  -&fe_indices[0] ;
1242  return_range.second =
1243  std::lower_bound(&fe_indices[0] + return_range.first,
1244  &fe_indices[0] + range.second,
1245  fe_index + 1)-&fe_indices[0];
1246  Assert(return_range.first >= range.first &&
1247  return_range.second <= range.second, ExcInternalError());
1248  return return_range;
1249  }
1250 }
