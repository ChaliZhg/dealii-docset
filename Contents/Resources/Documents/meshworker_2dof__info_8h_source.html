292  {
293  indices.resize(c->get_fe().dofs_per_cell);

295  c->get_active_or_mg_dof_indices(indices);

453  template <int dim, class DOFINFO>
454  template <class ASSEMBLER>
455  inline void
456  DoFInfoBox<dim, DOFINFO>::assemble (ASSEMBLER &assembler) const
457  {
458  if (!cell_valid)
459  return;
460 
461  assembler.assemble(cell);
462  for (unsigned int i=0; i<GeometryInfo<dim>::faces_per_cell; ++i)
463  {
464  // Only do something if data available
465  if (interior_face_available[i])
466  {
467  // If both data
468  // available, it is an
469  // interior face
470  if (exterior_face_available[i])
471  assembler.assemble(interior[i], exterior[i]);
472  else
473  assembler.assemble(interior[i]);
474  }
475  }
476  }
