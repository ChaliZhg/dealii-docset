159 template <int dim, int spacedim, int chartdim, int intermediate_dim,
160  int dim1, int dim2>
161 DerivativeForm<1,chartdim,spacedim>
162 CompositionManifold<dim,spacedim,chartdim,intermediate_dim,dim1,dim2>::push_forward_gradient(const Point<chartdim> &chart_point) const
163 {
164  DerivativeForm<1,chartdim,intermediate_dim> DF =
165  F->push_forward_gradient(chart_point);
166 
167  DerivativeForm<1,intermediate_dim,spacedim> DG =
168  G->push_forward_gradient(F->push_forward(chart_point));
169 
170  DerivativeForm<1,chartdim,spacedim> DF_DG;
171 
172  for (unsigned int d=0; d<spacedim; ++d)
173  for (unsigned int c=0; c<chartdim; ++c)
174  for (unsigned int s=0; s<intermediate_dim; ++s)
175  DF_DG[d][c] += DG[d][s]*DF[s][c];
176 
177  return DF_DG;
178 }
