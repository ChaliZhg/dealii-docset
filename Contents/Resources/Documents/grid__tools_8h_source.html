1715  template <int dim, typename Predicate, int spacedim>
1716  void transform (const Predicate &predicate,
1717  Triangulation<dim, spacedim> &triangulation)
1718  {
1719  std::vector<bool> treated_vertices (triangulation.n_vertices(),
1720  false);
1721 
1722  // loop over all active cells, and
1723  // transform those vertices that
1724  // have not yet been touched. note
1725  // that we get to all vertices in
1726  // the triangulation by only
1727  // visiting the active cells.
1728  typename Triangulation<dim, spacedim>::active_cell_iterator
1729  cell = triangulation.begin_active (),
1730  endc = triangulation.end ();
1731  for (; cell!=endc; ++cell)
1732  for (unsigned int v=0; v<GeometryInfo<dim>::vertices_per_cell; ++v)
1733  if (treated_vertices[cell->vertex_index(v)] == false)
1734  {
1735  // transform this vertex
1736  cell->vertex(v) = predicate(cell->vertex(v));
1737  // and mark it as treated
1738  treated_vertices[cell->vertex_index(v)] = true;
1739  };
1740 
1741 
1742  // now fix any vertices on hanging nodes so that we don't create any holes
1743  if (dim==2)
1744  {
1745  typename Triangulation<dim,spacedim>::active_cell_iterator
1746  cell = triangulation.begin_active(),
1747  endc = triangulation.end();
1748  for (; cell!=endc; ++cell)
1749  for (unsigned int face=0; face<GeometryInfo<dim>::faces_per_cell; ++face)
1750  if (cell->face(face)->has_children() &&
1751  !cell->face(face)->at_boundary())
1752  {
1753  // this line has children
1754  cell->face(face)->child(0)->vertex(1)
1755  = (cell->face(face)->vertex(0) +
1756  cell->face(face)->vertex(1)) / 2;
1757  }
1758  }
1759  else if (dim==3)
1760  {
1761  typename Triangulation<dim,spacedim>::active_cell_iterator
1762  cell = triangulation.begin_active(),
1763  endc = triangulation.end();
1764  for (; cell!=endc; ++cell)
1765  for (unsigned int face=0; face<GeometryInfo<dim>::faces_per_cell; ++face)
1766  if (cell->face(face)->has_children() &&
1767  !cell->face(face)->at_boundary())
1768  {
1769  // this face has hanging nodes
1770  cell->face(face)->child(0)->vertex(1)
1771  = (cell->face(face)->vertex(0) + cell->face(face)->vertex(1)) / 2.0;
1772  cell->face(face)->child(0)->vertex(2)
1773  = (cell->face(face)->vertex(0) + cell->face(face)->vertex(2)) / 2.0;
1774  cell->face(face)->child(1)->vertex(3)
1775  = (cell->face(face)->vertex(1) + cell->face(face)->vertex(3)) / 2.0;
1776  cell->face(face)->child(2)->vertex(3)
1777  = (cell->face(face)->vertex(2) + cell->face(face)->vertex(3)) / 2.0;
1778 
1779  // center of the face
1780  cell->face(face)->child(0)->vertex(3)
1781  = (cell->face(face)->vertex(0) + cell->face(face)->vertex(1)
1782  + cell->face(face)->vertex(2) + cell->face(face)->vertex(3)) / 4.0;
1783  }
1784  }
1785 
1786  // Make sure FEValues notices that the mesh has changed
1787  triangulation.signals.mesh_movement();
1788  }

1792  template <class MeshType>
1793  std::vector<typename MeshType::active_cell_iterator>
1794  get_active_child_cells (const typename MeshType::cell_iterator &cell)
1795  {
1796  std::vector<typename MeshType::active_cell_iterator> child_cells;
1797 
1798  if (cell->has_children())
1799  {
1800  for (unsigned int child=0;
1801  child<cell->n_children(); ++child)
1802  if (cell->child (child)->has_children())
1803  {
1804  const std::vector<typename MeshType::active_cell_iterator>
1805  children = get_active_child_cells<MeshType> (cell->child(child));
1806  child_cells.insert (child_cells.end(),
1807  children.begin(), children.end());
1808  }
1809  else
1810  child_cells.push_back (cell->child(child));
1811  }
1812 
1813  return child_cells;
1814  }

1818  template <class MeshType>
1819  void
1820  get_active_neighbors(const typename MeshType::active_cell_iterator &cell,
1821  std::vector<typename MeshType::active_cell_iterator> &active_neighbors)
1822  {
1823  active_neighbors.clear ();
1824  for (unsigned int n=0; n<GeometryInfo<MeshType::dimension>::faces_per_cell; ++n)
1825  if (! cell->at_boundary(n))
1826  {
1827  if (MeshType::dimension == 1)
1828  {
1829  // check children of neighbor. note
1830  // that in 1d children of the neighbor
1831  // may be further refined. In 1d the
1832  // case is simple since we know what
1833  // children bound to the present cell
1834  typename MeshType::cell_iterator
1835  neighbor_child = cell->neighbor(n);
1836  if (!neighbor_child->active())
1837  {
1838  while (neighbor_child->has_children())
1839  neighbor_child = neighbor_child->child (n==0 ? 1 : 0);
1840 
1841  Assert (neighbor_child->neighbor(n==0 ? 1 : 0)==cell,
1842  ExcInternalError());
1843  }
1844  active_neighbors.push_back (neighbor_child);
1845  }
1846  else
1847  {
1848  if (cell->face(n)->has_children())
1849  // this neighbor has children. find
1850  // out which border to the present
1851  // cell
1852  for (unsigned int c=0; c<cell->face(n)->number_of_children(); ++c)
1853  active_neighbors.push_back (cell->neighbor_child_on_subface(n,c));
1854  else
1855  {
1856  // the neighbor must be active
1857  // himself
1858  Assert(cell->neighbor(n)->active(), ExcInternalError());
1859  active_neighbors.push_back(cell->neighbor(n));
1860  }
1861  }
1862  }
1863  }
1864 }
