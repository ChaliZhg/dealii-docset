75 template <int dim, int spacedim>
76 std::string
77 FE_Q<dim,spacedim>::get_name () const
78 {
79  // note that the FETools::get_fe_by_name function depends on the
80  // particular format of the string this function returns, so they have to be
81  // kept in synch
82 
83  std::ostringstream namebuf;
84  bool equidistant = true;
85  std::vector<double> points(this->degree+1);
86 
87  // Decode the support points in one coordinate direction.
88  std::vector<unsigned int> lexicographic = this->poly_space.get_numbering_inverse();
89  for (unsigned int j=0; j<=this->degree; j++)
90  points[j] = this->unit_support_points[lexicographic[j]][0];
91 
92  // Check whether the support points are equidistant.
93  for (unsigned int j=0; j<=this->degree; j++)
94  if (std::fabs(points[j] - (double)j/this->degree) > 1e-15)
95  {
96  equidistant = false;
97  break;
98  }
99 
100  if (equidistant == true)
101  {
102  if (this->degree > 2)
103  namebuf << "FE_Q<"
104  << Utilities::dim_string(dim,spacedim)
105  << ">(QIterated(QTrapez()," << this->degree << "))";
106  else
107  namebuf << "FE_Q<"
108  << Utilities::dim_string(dim,spacedim)
109  << ">(" << this->degree << ")";
110  }
111  else
112  {
113  // Check whether the support points come from QGaussLobatto.
114  const QGaussLobatto<1> points_gl(this->degree+1);
115  bool gauss_lobatto = true;
116  for (unsigned int j=0; j<=this->degree; j++)
117  if (points[j] != points_gl.point(j)(0))
118  {
119  gauss_lobatto = false;
120  break;
121  }
122  if (gauss_lobatto == true)
123  namebuf << "FE_Q<"
124  << Utilities::dim_string(dim,spacedim)
125  << ">(" << this->degree << ")";
126  else
127  namebuf << "FE_Q<"
128  << Utilities::dim_string(dim,spacedim)
129  << ">(QUnknownNodes(" << this->degree << "))";
130  }
131  return namebuf.str();
132 }

138 FE_Q<dim,spacedim>::
139 convert_generalized_support_point_values_to_nodal_values (const std::vector<Vector<double> > &support_point_values,
140  std::vector<double> &nodal_values) const
141 {
142  AssertDimension (support_point_values.size(),
143  this->get_unit_support_points().size());
144  AssertDimension (support_point_values.size(),
145  nodal_values.size());
146  AssertDimension (this->dofs_per_cell,
147  nodal_values.size());
148 
149  for (unsigned int i=0; i<this->dofs_per_cell; ++i)
150  {
151  AssertDimension (support_point_values[i].size(), 1);
152 
153  nodal_values[i] = support_point_values[i](0);
154  }
155 }
