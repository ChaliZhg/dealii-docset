31  // fill_copy_indices(). It represents an entry in the copy_indices* map,
32  // that associates a level dof index with a global dof index.
33  struct DoFPair
34  {
35  unsigned int level;
36  types::global_dof_index global_dof_index;
37  types::global_dof_index level_dof_index;

40  const types::global_dof_index global_dof_index,
41  const types::global_dof_index level_dof_index)
42  :
43  level(level),
44  global_dof_index(global_dof_index),
45  level_dof_index(level_dof_index)
46  {}

56  // Internal function for filling the copy indices from global to level
57  // indices
58  template <int dim, int spacedim>
59  void fill_copy_indices(const ::DoFHandler<dim,spacedim> &mg_dof,
60  const MGConstrainedDoFs *mg_constrained_dofs,
61  std::vector<std::vector<std::pair<types::global_dof_index, types::global_dof_index> > > &copy_indices,
62  std::vector<std::vector<std::pair<types::global_dof_index, types::global_dof_index> > > &copy_indices_global_mine,
63  std::vector<std::vector<std::pair<types::global_dof_index, types::global_dof_index> > > &copy_indices_level_mine)
64  {
65  // Now we are filling the variables copy_indices*, which are essentially
66  // maps from global to mgdof for each level stored as a std::vector of
67  // pairs. We need to split this map on each level depending on the
68  // ownership of the global and mgdof, so that we later not access
69  // non-local elements in copy_to/from_mg.
70  // We keep track in the bitfield dof_touched which global dof has been
71  // processed already (on the current level). This is the same as the
72  // multigrid running in serial.
73 
74  // map cpu_index -> vector of data
75  // that will be copied into copy_indices_level_mine
76  std::vector<DoFPair> send_data_temp;
77 
78  const unsigned int n_levels = mg_dof.get_triangulation().n_global_levels();
79  copy_indices.resize(n_levels);
80  copy_indices_global_mine.resize(n_levels);
81  copy_indices_level_mine.resize(n_levels);
82  IndexSet globally_relevant;
83  DoFTools::extract_locally_relevant_dofs(mg_dof, globally_relevant);
84 
85  const unsigned int dofs_per_cell = mg_dof.get_fe().dofs_per_cell;
86  std::vector<types::global_dof_index> global_dof_indices (dofs_per_cell);
87  std::vector<types::global_dof_index> level_dof_indices (dofs_per_cell);
88 
89  for (unsigned int level=0; level<n_levels; ++level)
90  {
91  std::vector<bool> dof_touched(globally_relevant.n_elements(), false);
92  copy_indices[level].clear();
93  copy_indices_level_mine[level].clear();
94  copy_indices_global_mine[level].clear();
95 
96  typename ::DoFHandler<dim,spacedim>::active_cell_iterator
97  level_cell = mg_dof.begin_active(level);
98  const typename ::DoFHandler<dim,spacedim>::active_cell_iterator
99  level_end = mg_dof.end_active(level);
100 
101  for (; level_cell!=level_end; ++level_cell)
102  {
103  if (mg_dof.get_triangulation().locally_owned_subdomain()!=numbers::invalid_subdomain_id
104  && (level_cell->level_subdomain_id()==numbers::artificial_subdomain_id
105  || level_cell->subdomain_id()==numbers::artificial_subdomain_id)
106  )
107  continue;
108 
109  // get the dof numbers of this cell for the global and the
110  // level-wise numbering
111  level_cell->get_dof_indices (global_dof_indices);
112  level_cell->get_mg_dof_indices (level_dof_indices);
113 
114  for (unsigned int i=0; i<dofs_per_cell; ++i)
115  {
116  // we need to ignore if the DoF is on a refinement edge (hanging node)
117  if (mg_constrained_dofs != nullptr
118  && mg_constrained_dofs->at_refinement_edge(level, level_dof_indices[i]))
119  continue;
120  types::global_dof_index global_idx = globally_relevant.index_within_set(global_dof_indices[i]);
121  //skip if we did this global dof already (on this or a coarser level)
122  if (dof_touched[global_idx])
123  continue;
124  bool global_mine = mg_dof.locally_owned_dofs().is_element(global_dof_indices[i]);
125  bool level_mine = mg_dof.locally_owned_mg_dofs(level).is_element(level_dof_indices[i]);
126 
127 
128  if (global_mine && level_mine)
129  {
130  copy_indices[level].emplace_back(global_dof_indices[i],
131  level_dof_indices[i]);
132  }
133  else if (global_mine)
134  {
135  copy_indices_global_mine[level].emplace_back(global_dof_indices[i],
136  level_dof_indices[i]);
137 
138  //send this to the owner of the level_dof:
139  send_data_temp.emplace_back(level,
140  global_dof_indices[i],
141  level_dof_indices[i]);
142  }
143  else
144  {
145  // somebody will send those to me
146  }
147 
148  dof_touched[global_idx] = true;
149  }
150  }
151  }
152 
153  const ::parallel::distributed::Triangulation<dim,spacedim> *tria =
154  (dynamic_cast<const parallel::distributed::Triangulation<dim,spacedim>*>
155  (&mg_dof.get_triangulation()));
156  AssertThrow(send_data_temp.size()==0 || tria!=nullptr, ExcMessage("parallel Multigrid only works with a distributed Triangulation!"));
157 
158 #ifdef DEAL_II_WITH_MPI
159  if (tria)
160  {
161  // TODO: Searching the owner for every single DoF becomes quite
162  // inefficient. Please fix this, Timo.
163  // The list of neighbors is symmetric (our neighbors have us as a
164  // neighbor), so we can use it to send and to know how many messages
165  // we will get.
166  std::set<types::subdomain_id> neighbors = tria->level_ghost_owners();
167  std::map<int, std::vector<DoFPair> > send_data;
168 
169  // * find owners of the level dofs and insert into send_data accordingly
170  for (typename std::vector<DoFPair>::iterator dofpair=send_data_temp.begin(); dofpair != send_data_temp.end(); ++dofpair)
171  {
172  std::set<types::subdomain_id>::iterator it;
173  for (it = neighbors.begin(); it != neighbors.end(); ++it)
174  {
175  if (mg_dof.locally_owned_mg_dofs_per_processor(dofpair->level)[*it].is_element(dofpair->level_dof_index))
176  {
177  send_data[*it].push_back(*dofpair);
178  break;
179  }
180  }
181  // Is this level dof not owned by any of our neighbors? That would
182  // certainly be a bug!
183  Assert(it!=neighbors.end(), ExcMessage("could not find DoF owner."));
184  }
185 
186  // * send
187  std::vector<MPI_Request> requests;
188  {
189  for (std::set<types::subdomain_id>::iterator it = neighbors.begin(); it != neighbors.end(); ++it)
190  {
191  requests.push_back(MPI_Request());
192  unsigned int dest = *it;
193  std::vector<DoFPair> &data = send_data[dest];
194  // If there is nothing to send, we still need to send a message,
195  // because the receiving end will be waitng. In that case we
196  // just send an empty message.
197  if (data.size())
198  {
199  const int ierr = MPI_Isend(&data[0], data.size()*sizeof(data[0]),
200  MPI_BYTE, dest, 71, tria->get_communicator(),
201  &*requests.rbegin());
202  AssertThrowMPI(ierr);
203  }
204  else
205  {
206  const int ierr = MPI_Isend(nullptr, 0, MPI_BYTE, dest, 71,
207  tria->get_communicator(), &*requests.rbegin());
208  AssertThrowMPI(ierr);
209  }
210  }
211  }
212 
213  // * receive
214  {
215  // We should get one message from each of our neighbors
216  std::vector<DoFPair> receive_buffer;
217  for (unsigned int counter=0; counter<neighbors.size(); ++counter)
218  {
219  MPI_Status status;
220  int len;
221  int ierr = MPI_Probe(MPI_ANY_SOURCE, 71, tria->get_communicator(), &status);
222  AssertThrowMPI(ierr);
223  ierr = MPI_Get_count(&status, MPI_BYTE, &len);
224  AssertThrowMPI(ierr);
225 
226  if (len==0)
227  {
228  ierr = MPI_Recv(nullptr, 0, MPI_BYTE, status.MPI_SOURCE, status.MPI_TAG,
229  tria->get_communicator(), &status);
230  AssertThrowMPI(ierr);
231  continue;
232  }
233 
234  int count = len / sizeof(DoFPair);
235  Assert(static_cast<int>(count * sizeof(DoFPair)) == len, ExcInternalError());
236  receive_buffer.resize(count);
237 
238  void *ptr = &receive_buffer[0];
239  ierr = MPI_Recv(ptr, len, MPI_BYTE, status.MPI_SOURCE, status.MPI_TAG,
240  tria->get_communicator(), &status);
241  AssertThrowMPI(ierr);
242 
243  for (unsigned int i=0; i<receive_buffer.size(); ++i)
244  {
245  copy_indices_level_mine[receive_buffer[i].level].push_back(
246  std::make_pair (receive_buffer[i].global_dof_index, receive_buffer[i].level_dof_index)
247  );
248  }
249  }
250  }
251 
252  // * wait for all MPI_Isend to complete
253  if (requests.size() > 0)
254  {
255  const int ierr = MPI_Waitall(requests.size(), &requests[0], MPI_STATUSES_IGNORE);
256  AssertThrowMPI(ierr);
257  requests.clear();
258  }
259 #ifdef DEBUG
260  // Make sure in debug mode, that everybody sent/received all packages
261  // on this level. If a deadlock occurs here, the list of expected
262  // senders is not computed correctly.
263  const int ierr = MPI_Barrier(tria->get_communicator());
264  AssertThrowMPI(ierr);
265 #endif
266  }
267 #endif
268 
269  // Sort the indices. This will produce more reliable debug output for
270  // regression tests and likely won't hurt performance even in release
271  // mode.
272  std::less<std::pair<types::global_dof_index, types::global_dof_index> > compare;
273  for (unsigned int level=0; level<copy_indices.size(); ++level)
274  std::sort(copy_indices[level].begin(), copy_indices[level].end(), compare);
275  for (unsigned int level=0; level<copy_indices_level_mine.size(); ++level)
276  std::sort(copy_indices_level_mine[level].begin(), copy_indices_level_mine[level].end(), compare);
277  for (unsigned int level=0; level<copy_indices_global_mine.size(); ++level)
278  std::sort(copy_indices_global_mine[level].begin(), copy_indices_global_mine[level].end(), compare);
279  }

283  // initialize the vectors needed for the transfer (and merge with the
284  // content in copy_indices_global_mine)
285  template <typename Number>
286  void
287  reinit_ghosted_vector(const IndexSet &locally_owned,
288  std::vector<types::global_dof_index> &ghosted_level_dofs,
289  const MPI_Comm &communicator,
290  LinearAlgebra::distributed::Vector<Number> &ghosted_level_vector,
291  std::vector<std::pair<unsigned int,unsigned int> > &copy_indices_global_mine)
292  {
293  std::sort(ghosted_level_dofs.begin(), ghosted_level_dofs.end());
294  IndexSet ghosted_dofs(locally_owned.size());
295  ghosted_dofs.add_indices(ghosted_level_dofs.begin(),
296  std::unique(ghosted_level_dofs.begin(),
297  ghosted_level_dofs.end()));
298  ghosted_dofs.compress();
299 
300  // Add possible ghosts from the previous content in the vector
301  if (ghosted_level_vector.size() == locally_owned.size())
302  {
303  // shift the local number of the copy indices according to the new
304  // partitioner that we are going to use for the vector
305  const std::shared_ptr<const Utilities::MPI::Partitioner> part
306  = ghosted_level_vector.get_partitioner();
307  ghosted_dofs.add_indices(part->ghost_indices());
308  for (unsigned int i=0; i<copy_indices_global_mine.size(); ++i)
309  copy_indices_global_mine[i].second =
310  locally_owned.n_elements() +
311  ghosted_dofs.index_within_set(part->local_to_global(copy_indices_global_mine[i].second));
312  }
313  ghosted_level_vector.reinit(locally_owned, ghosted_dofs, communicator);
314  }

316  // Transform the ghost indices to local index space for the vector
317  inline void
318  copy_indices_to_mpi_local_numbers(const Utilities::MPI::Partitioner &part,
319  const std::vector<types::global_dof_index> &mine,
320  const std::vector<types::global_dof_index> &remote,
321  std::vector<unsigned int> &localized_indices)
322  {
323  localized_indices.resize(mine.size()+remote.size(),
324  numbers::invalid_unsigned_int);
325  for (unsigned int i=0; i<mine.size(); ++i)
326  if (mine[i] != numbers::invalid_dof_index)
327  localized_indices[i] = part.global_to_local(mine[i]);
328 
329  for (unsigned int i=0; i<remote.size(); ++i)
330  if (remote[i] != numbers::invalid_dof_index)
331  localized_indices[i+mine.size()] = part.global_to_local(remote[i]);
332  }

334  // given the collection of child cells in lexicographic ordering as seen
335  // from the parent, compute the first index of the given child
336  template <int dim>
337  unsigned int
338  compute_shift_within_children(const unsigned int child,
339  const unsigned int fe_shift_1d,
340  const unsigned int fe_degree)
341  {
342  // we put the degrees of freedom of all child cells in lexicographic
343  // ordering
344  unsigned int c_tensor_index[dim];
345  unsigned int tmp = child;
346  for (unsigned int d=0; d<dim; ++d)
347  {
348  c_tensor_index[d] = tmp % 2;
349  tmp /= 2;
350  }
351  const unsigned int n_child_dofs_1d = fe_degree + 1 + fe_shift_1d;
352  unsigned int factor = 1;
353  unsigned int shift = fe_shift_1d * c_tensor_index[0];
354  for (unsigned int d=1; d<dim; ++d)
355  {
356  factor *= n_child_dofs_1d;
357  shift = shift + factor * fe_shift_1d * c_tensor_index[d];
358  }
359  return shift;
360  }

362  // puts the indices on the given child cell in lexicographic ordering with
363  // respect to the collection of all child cells as seen from the parent
364  template <int dim>
365  void add_child_indices(const unsigned int child,
366  const unsigned int fe_shift_1d,
367  const unsigned int fe_degree,
368  const std::vector<unsigned int> &lexicographic_numbering,
369  const std::vector<types::global_dof_index> &local_dof_indices,
370  types::global_dof_index *target_indices)
371  {
372  const unsigned int n_child_dofs_1d = fe_degree + 1 + fe_shift_1d;
373  const unsigned int shift =
374  compute_shift_within_children<dim>(child, fe_shift_1d, fe_degree);
375  const unsigned int n_components =
376  local_dof_indices.size()/Utilities::fixed_power<dim>(fe_degree+1);
377  types::global_dof_index *indices = target_indices + shift;
378  const unsigned int n_scalar_cell_dofs = Utilities::fixed_power<dim>(n_child_dofs_1d);
379  for (unsigned int c=0, m=0; c<n_components; ++c)
380  for (unsigned int k=0; k<(dim>2 ? (fe_degree+1) : 1); ++k)
381  for (unsigned int j=0; j<(dim>1 ? (fe_degree+1) : 1); ++j)
382  for (unsigned int i=0; i<(fe_degree+1); ++i, ++m)
383  {
384  const unsigned int index = c*n_scalar_cell_dofs+k*n_child_dofs_1d*
385  n_child_dofs_1d+j*n_child_dofs_1d+i;
386  Assert(indices[index] == numbers::invalid_dof_index ||
387  indices[index] == local_dof_indices[lexicographic_numbering[m]],
388  ExcInternalError());
389  indices[index] = local_dof_indices[lexicographic_numbering[m]];
390  }
391  }

393  template <int dim, typename Number>
394  void setup_element_info(ElementInfo<Number> &elem_info,const FiniteElement<1> &fe,
395  const ::DoFHandler<dim> &mg_dof)
396  {
397  // currently, we have only FE_Q and FE_DGQ type elements implemented
398  elem_info.n_components = mg_dof.get_fe().element_multiplicity(0);
399  AssertDimension(Utilities::fixed_power<dim>(fe.dofs_per_cell)*elem_info.n_components,
400  mg_dof.get_fe().dofs_per_cell);
401  AssertDimension(fe.degree, mg_dof.get_fe().degree);
402  elem_info.fe_degree = fe.degree;
403  elem_info.element_is_continuous = fe.dofs_per_vertex > 0;
404  Assert(fe.dofs_per_vertex < 2, ExcNotImplemented());
405 
406  // step 1.2: get renumbering of 1D basis functions to lexicographic
407  // numbers. The distinction according to fe.dofs_per_vertex is to support
408  // both continuous and discontinuous bases.
409  std::vector<unsigned int> renumbering(fe.dofs_per_cell);
410  {
411  AssertIndexRange(fe.dofs_per_vertex, 2);
412  renumbering[0] = 0;
413  for (unsigned int i=0; i<fe.dofs_per_line; ++i)
414  renumbering[i+fe.dofs_per_vertex] =
415  GeometryInfo<1>::vertices_per_cell*fe.dofs_per_vertex + i;
416  if (fe.dofs_per_vertex > 0)
417  renumbering[fe.dofs_per_cell-fe.dofs_per_vertex] = fe.dofs_per_vertex;
418  }
419 
420  // step 1.3: create a dummy 1D quadrature formula to extract the
421  // lexicographic numbering for the elements
422  std::vector<Point<1> > basic_support_points = fe.get_unit_support_points();
423  Assert(fe.dofs_per_vertex == 0 || fe.dofs_per_vertex == 1,
424  ExcNotImplemented());
425  const unsigned int shift = fe.dofs_per_cell - fe.dofs_per_vertex;
426  const unsigned int n_child_dofs_1d = (fe.dofs_per_vertex > 0 ?
427  (2 * fe.dofs_per_cell - 1) :
428  (2 * fe.dofs_per_cell));
429 
430  elem_info.n_child_cell_dofs = elem_info.n_components*Utilities::fixed_power<dim>(n_child_dofs_1d);
431  const Quadrature<1> dummy_quadrature(std::vector<Point<1> >(1, Point<1>()));
432  internal::MatrixFreeFunctions::ShapeInfo<Number> shape_info;
433  shape_info.reinit(dummy_quadrature, mg_dof.get_fe(), 0);
434  elem_info.lexicographic_numbering = shape_info.lexicographic_numbering;
435 
436  // step 1.4: get the 1d prolongation matrix and combine from both children
437  elem_info.prolongation_matrix_1d.resize(fe.dofs_per_cell*n_child_dofs_1d);
438 
439  for (unsigned int c=0; c<GeometryInfo<1>::max_children_per_cell; ++c)
440  for (unsigned int i=0; i<fe.dofs_per_cell; ++i)
441  for (unsigned int j=0; j<fe.dofs_per_cell; ++j)
442  elem_info.prolongation_matrix_1d[i*n_child_dofs_1d+j+c*shift] =
443  fe.get_prolongation_matrix(c)(renumbering[j],renumbering[i]);
444  }

447  // Sets up most of the internal data structures of the MGTransferMatrixFree
448  // class
449  template <int dim, typename Number>
450  void setup_transfer(const ::DoFHandler<dim> &mg_dof,
451  const MGConstrainedDoFs *mg_constrained_dofs,
452  ElementInfo<Number> &elem_info,
453  std::vector<std::vector<unsigned int> > &level_dof_indices,
454  std::vector<std::vector<std::pair<unsigned int,unsigned int> > > &parent_child_connect,
455  std::vector<unsigned int> &n_owned_level_cells,
456  std::vector<std::vector<std::vector<unsigned short> > > &dirichlet_indices,
457  std::vector<std::vector<Number> > &weights_on_refined,
458  std::vector<std::vector<std::pair<unsigned int, unsigned int> > > &copy_indices_global_mine,
459  MGLevelObject<LinearAlgebra::distributed::Vector<Number> > &ghosted_level_vector)
460  {
461  level_dof_indices.clear();
462  parent_child_connect.clear();
463  n_owned_level_cells.clear();
464  dirichlet_indices.clear();
465  weights_on_refined.clear();
466 
467  // we collect all child DoFs of a mother cell together. For faster
468  // tensorized operations, we align the degrees of freedom
469  // lexicographically. We distinguish FE_Q elements and FE_DGQ elements
470 
471  const ::Triangulation<dim> &tria = mg_dof.get_triangulation();
472 
473  // ---------------------------- 1. Extract 1D info about the finite element
474  // step 1.1: create a 1D copy of the finite element from FETools where we
475  // substitute the template argument
476  AssertDimension(mg_dof.get_fe().n_base_elements(), 1);
477  std::string fe_name = mg_dof.get_fe().base_element(0).get_name();
478  {
479  const std::size_t template_starts = fe_name.find_first_of('<');
480  Assert (fe_name[template_starts+1] == (dim==1?'1':(dim==2?'2':'3')),
481  ExcInternalError());
482  fe_name[template_starts+1] = '1';
483  }
484  std::shared_ptr<FiniteElement<1> > fe_1d
485  (FETools::get_fe_by_name<1,1>(fe_name));
486  const FiniteElement<1> &fe = *fe_1d;
487 
488  setup_element_info(elem_info,fe,mg_dof);
489 
490 
491  // -------------- 2. Extract and match dof indices between child and parent
492  const unsigned int n_levels = tria.n_global_levels();
493  level_dof_indices.resize(n_levels);
494  parent_child_connect.resize(n_levels-1);
495  n_owned_level_cells.resize(n_levels-1);
496  std::vector<std::vector<unsigned int> > coarse_level_indices(n_levels-1);
497  for (unsigned int level=0; level<std::min(tria.n_levels(),n_levels-1); ++level)
498  coarse_level_indices[level].resize(tria.n_raw_cells(level),
499  numbers::invalid_unsigned_int);
500  std::vector<types::global_dof_index> local_dof_indices(mg_dof.get_fe().dofs_per_cell);
501  dirichlet_indices.resize(n_levels-1);
502 
503  // We use the vectors stored ghosted_level_vector in the base class for
504  // keeping ghosted transfer indices. To avoid keeping two very similar
505  // vectors, we merge them here.
506  if (ghosted_level_vector.max_level() != n_levels-1)
507  ghosted_level_vector.resize(0, n_levels-1);
508 
509  for (unsigned int level=n_levels-1; level > 0; --level)
510  {
511  unsigned int counter = 0;
512  std::vector<types::global_dof_index> global_level_dof_indices;
513  std::vector<types::global_dof_index> global_level_dof_indices_remote;
514  std::vector<types::global_dof_index> ghosted_level_dofs;
515  std::vector<types::global_dof_index> global_level_dof_indices_l0;
516  std::vector<types::global_dof_index> ghosted_level_dofs_l0;
517 
518  // step 2.1: loop over the cells on the coarse side
519  typename ::DoFHandler<dim>::cell_iterator cell, endc = mg_dof.end(level-1);
520  for (cell = mg_dof.begin(level-1); cell != endc; ++cell)
521  {
522  // need to look into a cell if it has children and it is locally
523  // owned
524  if (!cell->has_children())
525  continue;

527  bool consider_cell = false;
528  if (tria.locally_owned_subdomain()==numbers::invalid_subdomain_id
529  || cell->level_subdomain_id()==tria.locally_owned_subdomain()
530  )
531  consider_cell = true;

534  // we also need to add the DoF indices for coarse cells where we
535  // own at least one child
536  bool cell_is_remote = !consider_cell;
537  for (unsigned int c=0; c<GeometryInfo<dim>::max_children_per_cell; ++c)
538  if (cell->child(c)->level_subdomain_id()==tria.locally_owned_subdomain())
539  {
540  consider_cell = true;
541  break;
542  }
543 
544  if (!consider_cell)
545  continue;
546 
547  // step 2.2: loop through children and append the dof indices to
548  // the appropriate list. We need separate lists for the owned
549  // coarse cell case (which will be part of
550  // restriction/prolongation between level-1 and level) and the
551  // remote case (which needs to store DoF indices for the
552  // operations between level and level+1).
553  AssertDimension(cell->n_children(),
554  GeometryInfo<dim>::max_children_per_cell);
555  std::vector<types::global_dof_index> &next_indices =
556  cell_is_remote ? global_level_dof_indices_remote : global_level_dof_indices;
557  const std::size_t start_index = next_indices.size();
558  next_indices.resize(start_index + elem_info.n_child_cell_dofs,
559  numbers::invalid_dof_index);

561  {

563  tria.locally_owned_subdomain())
564  continue;
565  cell->child(c)->get_mg_dof_indices(local_dof_indices);

567  const IndexSet &owned_level_dofs = mg_dof.locally_owned_mg_dofs(level);
568  for (unsigned int i=0; i<local_dof_indices.size(); ++i)
569  if (!owned_level_dofs.is_element(local_dof_indices[i]))
570  ghosted_level_dofs.push_back(local_dof_indices[i]);
571 
572  add_child_indices<dim>(c, fe.dofs_per_cell - fe.dofs_per_vertex,
573  fe.degree, elem_info.lexicographic_numbering,
574  local_dof_indices,
575  &next_indices[start_index]);
576 
577  // step 2.3 store the connectivity to the parent
578  if (cell->child(c)->has_children() &&
579  (tria.locally_owned_subdomain()==numbers::invalid_subdomain_id
580  || cell->child(c)->level_subdomain_id()==tria.locally_owned_subdomain()
581  ))
582  {

584  AssertIndexRange(child_index, parent_child_connect[level].size());
585  unsigned int parent_index = counter;
586  // remote cells, i.e., cells where we work on a further
587  // level but are not treated on the current level, need to
588  // be placed at the end of the list; however, we do not
589  // yet know the exact position in the array, so shift
590  // their parent index by the number of cells so we can set
591  // the correct number after the end of this loop
592  if (cell_is_remote)
593  parent_index = start_index/elem_info.n_child_cell_dofs + tria.n_cells(level);
594  parent_child_connect[level][child_index] =
595  std::make_pair(parent_index, c);
596  AssertIndexRange(mg_dof.get_fe().dofs_per_cell,
597  static_cast<unsigned short>(-1));
598 
599  // set Dirichlet boundary conditions (as a list of
600  // constrained DoFs) for the child
601  if (mg_constrained_dofs != nullptr)
602  for (unsigned int i=0; i<mg_dof.get_fe().dofs_per_cell; ++i)
603  if (mg_constrained_dofs->is_boundary_index(level,
604  local_dof_indices[elem_info.lexicographic_numbering[i]]))
605  dirichlet_indices[level][child_index].push_back(i);
606  }
607  }
608  if (!cell_is_remote)
609  {
610  AssertIndexRange(static_cast<unsigned int>(cell->index()),
611  coarse_level_indices[level-1].size());
612  coarse_level_indices[level-1][cell->index()] = counter++;
613  }
614 
615  // step 2.4: include indices for the coarsest cells. we still
616  // insert the indices as if they were from a child in order to use
617  // the same code (the coarsest level does not matter much in terms
618  // of memory, so we gain in code simplicity)

620  {

622

624  for (unsigned int i=0; i<local_dof_indices.size(); ++i)

626  ghosted_level_dofs_l0.push_back(local_dof_indices[i]);

629  global_level_dof_indices_l0.resize(start_index+elem_info.n_child_cell_dofs,
630  numbers::invalid_dof_index);

632  fe.degree, elem_info.lexicographic_numbering,
633  local_dof_indices,
634  &global_level_dof_indices_l0[start_index]);
635

638  for (unsigned int i=0; i<mg_dof.get_fe().dofs_per_cell; ++i)
639  if (mg_constrained_dofs->is_boundary_index(0, local_dof_indices[elem_info.lexicographic_numbering[i]]))
640  dirichlet_indices[0].back().push_back(i);
641  }
642  }
643 
644  // step 2.5: store information about the current level and prepare the
645  // Dirichlet indices and parent-child relationship for the next
646  // coarser level
647  AssertDimension(counter*elem_info.n_child_cell_dofs, global_level_dof_indices.size());
648  n_owned_level_cells[level-1] = counter;
649  dirichlet_indices[level-1].resize(counter);
650  parent_child_connect[level-1].

652  numbers::invalid_unsigned_int));
653 
654  // step 2.6: put the cells with remotely owned parent to the end of
655  // the list (these are needed for the transfer from level to level+1
656  // but not for the transfer from level-1 to level).
657  if (level < n_levels-1)
658  for (std::vector<std::pair<unsigned int,unsigned int> >::iterator
659  i=parent_child_connect[level].begin(); i!=parent_child_connect[level].end(); ++i)
660  if (i->first >= tria.n_cells(level))
661  {
662  i->first -= tria.n_cells(level);
663  i->first += counter;
664  }
665 
666  // step 2.7: Initialize the ghosted vector
667  const parallel::Triangulation<dim,dim> *ptria =
668  (dynamic_cast<const parallel::Triangulation<dim,dim>*> (&tria));
669  const MPI_Comm communicator =
670  ptria != nullptr ? ptria->get_communicator() : MPI_COMM_SELF;
671 
672  reinit_ghosted_vector (mg_dof.locally_owned_mg_dofs(level),
673  ghosted_level_dofs, communicator,
674  ghosted_level_vector[level],
675  copy_indices_global_mine[level]);
676 
677  copy_indices_to_mpi_local_numbers(*ghosted_level_vector[level].get_partitioner(),
678  global_level_dof_indices,
679  global_level_dof_indices_remote,
680  level_dof_indices[level]);
681  // step 2.8: Initialize the ghosted vector for level 0
682  if (level == 1)
683  {
684  for (unsigned int i = 0; i<parent_child_connect[0].size(); ++i)
685  parent_child_connect[0][i] = std::make_pair(i, 0U);
686 
687  reinit_ghosted_vector (mg_dof.locally_owned_mg_dofs(0),
688  ghosted_level_dofs_l0, communicator,
689  ghosted_level_vector[0],
690  copy_indices_global_mine[0]);
691 
692  copy_indices_to_mpi_local_numbers(*ghosted_level_vector[0].get_partitioner(),
693  global_level_dof_indices_l0,
694  std::vector<types::global_dof_index>(),
695  level_dof_indices[0]);
696  }
697  }
698 
699  // ---------------------- 3. compute weights to make restriction additive
700 
701  const unsigned int n_child_dofs_1d = fe.degree + 1 + fe.dofs_per_cell - fe.dofs_per_vertex;
702 
703  // get the valence of the individual components and compute the weights as
704  // the inverse of the valence
705  weights_on_refined.resize(n_levels-1);
706  for (unsigned int level = 1; level<n_levels; ++level)
707  {
708  ghosted_level_vector[level] = 0;
709  for (unsigned int c=0; c<n_owned_level_cells[level-1]; ++c)
710  for (unsigned int j=0; j<elem_info.n_child_cell_dofs; ++j)
711  ghosted_level_vector[level].local_element(level_dof_indices[level][elem_info.n_child_cell_dofs*c+j]) += Number(1.);
712  ghosted_level_vector[level].compress(VectorOperation::add);
713  ghosted_level_vector[level].update_ghost_values();
714 
715  std::vector<unsigned int> degree_to_3 (n_child_dofs_1d);
716  degree_to_3[0] = 0;
717  for (unsigned int i=1; i<n_child_dofs_1d-1; ++i)
718  degree_to_3[i] = 1;
719  degree_to_3.back() = 2;
720 
721  // we only store 3^dim weights because all dofs on a line have the
722  // same valence, and all dofs on a quad have the same valence.
723  weights_on_refined[level-1].resize(n_owned_level_cells[level-1]*Utilities::fixed_power<dim>(3));
724  for (unsigned int c=0; c<n_owned_level_cells[level-1]; ++c)
725  for (unsigned int k=0, m=0; k<(dim>2 ? n_child_dofs_1d : 1); ++k)
726  for (unsigned int j=0; j<(dim>1 ? n_child_dofs_1d : 1); ++j)
727  {
728  unsigned int shift = 9*degree_to_3[k] + 3*degree_to_3[j];
729  for (unsigned int i=0; i<n_child_dofs_1d; ++i, ++m)
730  weights_on_refined[level-1][c*Utilities::fixed_power<dim>(3)+shift+degree_to_3[i]] = Number(1.)/
731  ghosted_level_vector[level].local_element(level_dof_indices[level][elem_info.n_child_cell_dofs*c+m]);
732  }
733  }
734
