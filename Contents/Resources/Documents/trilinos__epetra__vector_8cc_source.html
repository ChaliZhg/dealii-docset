93  Vector &Vector::operator= (const Vector &V)
94  {
95  // Distinguish three cases:
96  // - First case: both vectors have the same layout.
97  // - Second case: both vectors have the same size but different layout.
98  // - Third case: the vectors have different size.
99  if (vector->Map().SameAs(V.trilinos_vector().Map()))
100  *vector = V.trilinos_vector();
101  else
102  {
103  if (size()==V.size())
104  {
105  Epetra_Import data_exchange(vector->Map(), V.trilinos_vector().Map());
106 
107  const int ierr = vector->Import(V.trilinos_vector(), data_exchange, Insert);
108  Assert(ierr==0, ExcTrilinosError(ierr));
109  (void) ierr;
110  }
111  else
112  vector.reset(new Epetra_FEVector(V.trilinos_vector()));
113  }
114 
115  return *this;
116  }

130  void Vector::import(const ReadWriteVector<double> &V,
131  VectorOperation::values operation,
132  std::shared_ptr<const CommunicationPatternBase> communication_pattern)
133  {
134  // If no communication pattern is given, create one. Otherwsie, use the
135  // one given.
136  if (communication_pattern == nullptr)
137  {
138  // The first time import is called, a communication pattern is created.
139  // Check if the communication pattern already exists and if it can be
140  // reused.
141  if ((source_stored_elements.size() != V.get_stored_elements().size()) ||
142  ((source_stored_elements.size() == V.get_stored_elements().size()) &&
143  (source_stored_elements != V.get_stored_elements())))
144  {
145  create_epetra_comm_pattern(V.get_stored_elements(),
146  dynamic_cast<const Epetra_MpiComm &>(vector->Comm()).Comm());
147  }
148  }
149  else
150  {
151  epetra_comm_pattern =
152  std::dynamic_pointer_cast<const CommunicationPattern> (communication_pattern);
153  AssertThrow(epetra_comm_pattern != nullptr,
154  ExcMessage(std::string("The communication pattern is not of type ") +
155  "LinearAlgebra::EpetraWrappers::CommunicationPattern."));
156  }
157 
158  Epetra_Import import(epetra_comm_pattern->get_epetra_import());
159 
160  // The TargetMap and the SourceMap have their roles inverted.
161  Epetra_FEVector source_vector(import.TargetMap());
162  double *values = source_vector.Values();
163  std::copy(V.begin(), V.end(), values);
164 
165  if (operation==VectorOperation::insert)
166  vector->Export(source_vector, import, Insert);
167  else
168  vector->Export(source_vector, import, Add);
169  }

194  Vector &Vector::operator+= (const VectorSpaceVector<double> &V)
195  {
196  // Check that casting will work.
197  Assert(dynamic_cast<const Vector *>(&V)!=nullptr, ExcVectorTypeNotCompatible());
198 
199  // Downcast V. If fails, throws an exception.
200  const Vector &down_V = dynamic_cast<const Vector &>(V);
201  // If the maps are the same we can Update right away.
202  if (vector->Map().SameAs(down_V.trilinos_vector().Map()))
203  {
204  const int ierr = vector->Update(1., down_V.trilinos_vector(), 1.);
205  Assert(ierr==0, ExcTrilinosError(ierr));
206  (void) ierr;
207  }
208  else
209  {
210  Assert(this->size()==down_V.size(),
211  ExcDimensionMismatch(this->size(), down_V.size()));
212 
213 #if DEAL_II_TRILINOS_VERSION_GTE(11,11,0)
214  Epetra_Import data_exchange (vector->Map(), down_V.trilinos_vector().Map());
215  const int ierr = vector->Import(down_V.trilinos_vector(),
216  data_exchange, Epetra_AddLocalAlso);
217  Assert(ierr==0, ExcTrilinosError(ierr));
218  (void) ierr;
219 #else
220  // In versions older than 11.11 the Import function is broken for adding
221  // Hence, we provide a workaround in this case
222 
223  Epetra_MultiVector dummy(vector->Map(), 1, false);
224  Epetra_Import data_exchange(dummy.Map(), down_V.trilinos_vector().Map());
225 
226  int ierr = dummy.Import(down_V.trilinos_vector(), data_exchange, Insert);
227  Assert(ierr==0, ExcTrilinosError(ierr));
228 
229  ierr = vector->Update(1.0, dummy, 1.0);
230  Assert(ierr==0, ExcTrilinosError(ierr));
231  (void) ierr;
232 #endif
233  }
234 
235  return *this;
236  }

368  {
369  // Check that casting will work.

470  ::IndexSet Vector::locally_owned_elements() const
471  {
472  IndexSet is (size());
473 
474  // easy case: local range is contiguous
475  if (vector->Map().LinearMap())
476  {
477 #ifndef DEAL_II_WITH_64BIT_INDICES
478  is.add_range(vector->Map().MinMyGID(), vector->Map().MaxMyGID()+1);
479 #else
480  is.add_range(vector->Map().MinMyGID64(), vector->Map().MaxMyGID64()+1);
481 #endif
482  }
483  else if (vector->Map().NumMyElements() > 0)
484  {
485  const size_type n_indices = vector->Map().NumMyElements();
486 #ifndef DEAL_II_WITH_64BIT_INDICES
487  unsigned int *vector_indices = (unsigned int *)vector->Map().MyGlobalElements();
488 #else
489  size_type *vector_indices = (size_type *)vector->Map().MyGlobalElements64();
490 #endif
491  is.add_indices(vector_indices, vector_indices+n_indices);
492  }
493  is.compress();
494 
495  return is;
496  }

514  void Vector::print(std::ostream &out,
515  const unsigned int precision,
516  const bool scientific,
517  const bool across) const
518  {
519  AssertThrow(out, ExcIO());
520  boost::io::ios_flags_saver restore_flags(out);
521 
522  // Get a representation of the vector and loop over all
523  // the elements
524  double *val;
525  int leading_dimension;
526  int ierr = vector->ExtractView(&val, &leading_dimension);
527 
528  Assert(ierr==0, ExcTrilinosError(ierr));
529  (void) ierr;
530  out.precision (precision);
531  if (scientific)
532  out.setf(std::ios::scientific, std::ios::floatfield);
533  else
534  out.setf(std::ios::fixed, std::ios::floatfield);
535 
536  if (across)
537  for (int i=0; i<vector->MyLength(); ++i)
538  out << val[i] << ' ';
539  else
540  for (int i=0; i<vector->MyLength(); ++i)
541  out << val[i] << std::endl;
542  out << std::endl;
543 
544  // restore the representation
545  // of the vector
546  AssertThrow(out, ExcIO());
547  }
