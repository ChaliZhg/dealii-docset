53  void handle_std_exception (const std::exception &exc)
54  {
55  // lock the following context
56  // to ensure that we don't
57  // print things over each other
58  // if we have trouble from
59  // multiple threads. release
60  // the lock before calling
61  // std::abort, though
62  static Mutex mutex;
63  {
64  Mutex::ScopedLock lock(mutex);
65 
66  std::cerr << std::endl << std::endl
67  << "---------------------------------------------------------"
68  << std::endl

70  << "was thrown and not caught. Since exceptions do not\n"

232  std::vector<std::pair<unsigned int,unsigned int> >
233  split_interval (const unsigned int begin,
234  const unsigned int end,
235  const unsigned int n_intervals)
236  {
237  Assert (end >= begin, ExcInternalError());
238 
239  const unsigned int n_elements = end-begin;
240  const unsigned int n_elements_per_interval = n_elements / n_intervals;
241  const unsigned int residual = n_elements % n_intervals;
242 
243  std::vector<std::pair<unsigned int,unsigned int> > return_values (n_intervals);
244 
245  return_values[0].first = begin;
246  for (unsigned int i=0; i<n_intervals; ++i)
247  {
248  if (i != n_intervals-1)
249  {
250  return_values[i].second = (return_values[i].first
251  + n_elements_per_interval);
252  // distribute residual in
253  // division equally among
254  // the first few
255  // subintervals
256  if (i < residual)
257  ++return_values[i].second;
258  return_values[i+1].first = return_values[i].second;
259  }
260  else
261  return_values[i].second = end;
262  };
263  return return_values;
264  }
