62 template <>
63 QGauss<1>::QGauss (const unsigned int n)
64  :
65  Quadrature<1> (n)
66 {
67  if (n == 0)
68  return;
69 
70  const unsigned int m = (n+1)/2;
71 
72  // tolerance for the Newton
73  // iteration below. we need to make
74  // it adaptive since on some
75  // machines (for example PowerPC)
76  // long double is the same as
77  // double -- in that case we can
78  // only get to a certain multiple
79  // of the accuracy of double there,
80  // while on other machines we'd
81  // like to go further down
82  //
83  // the situation is complicated by
84  // the fact that even if long
85  // double exists and is described
86  // by std::numeric_limits, we may
87  // not actually get the additional
88  // precision. One case where this
89  // happens is on x86, where one can
90  // set hardware flags that disable
91  // long double precision even for
92  // long double variables. these
93  // flags are not usually set, but
94  // for example matlab sets them and
95  // this then breaks deal.II code
96  // that is run as a subroutine to
97  // matlab...
98  //
99  // a similar situation exists, btw,
100  // when running programs under
101  // valgrind up to and including at
102  // least version 3.3: valgrind's
103  // emulator only supports 64 bit
104  // arithmetic, even for 80 bit long
105  // doubles.
106  const long double
107  long_double_eps = static_cast<long double>(std::numeric_limits<long double>::epsilon()),
108  double_eps = static_cast<long double>(std::numeric_limits<double>::epsilon());
109 
110  // now check whether long double is more
111  // accurate than double, and set
112  // tolerances accordingly. generate a one
113  // that really is generated at run-time
114  // and is not optimized away by the
115  // compiler. that makes sure that the
116  // tolerance is set at run-time with the
117  // current behavior, not at compile-time
118  // (not doing so leads to trouble with
119  // valgrind for example).
120  volatile long double runtime_one = 1.0;
121  const long double tolerance
122  = (runtime_one + long_double_eps != runtime_one
123  ?
124  std::max (double_eps / 100, long_double_eps * 5)
125  :
126  double_eps * 5
127  );
128 
129 
130  for (unsigned int i=1; i<=m; ++i)
131  {
132  long double z = std::cos(numbers::PI * (i-.25)/(n+.5));
133 
134  long double pp;
135  long double p1;
136 
137  // Newton iteration
138  do
139  {
140  // compute L_n (z)
141  p1 = 1.;
142  long double p2 = 0.;
143  for (unsigned int j=0; j<n; ++j)
144  {
145  const long double p3 = p2;
146  p2 = p1;
147  p1 = ((2.*j+1.)*z*p2-j*p3)/(j+1);
148  }
149  pp = n*(z*p1-p2)/(z*z-1);
150  z = z-p1/pp;
151  }
152  while (std::abs(p1/pp) > tolerance);
153 
154  double x = .5*z;
155  this->quadrature_points[i-1] = Point<1>(.5-x);
156  this->quadrature_points[n-i] = Point<1>(.5+x);
157 
158  double w = 1./((1.-z*z)*pp*pp);
159  this->weights[i-1] = w;
160  this->weights[n-i] = w;
161  }
162 }

166 QGaussLobatto<1>::QGaussLobatto (const unsigned int n)
167  :
168  Quadrature<1> (n)
169 {
170  Assert (n >= 2, ExcNotImplemented());
171 
172  std::vector<long double> points = compute_quadrature_points(n, 1, 1);
173  std::vector<long double> w = compute_quadrature_weights(points, 0, 0);
174 
175  // scale points to the interval
176  // [0.0, 1.0]:
177  for (unsigned int i=0; i<points.size(); ++i)
178  {
179  this->quadrature_points[i] = Point<1>(0.5 + 0.5*static_cast<double>(points[i]));
180  this->weights[i] = 0.5*w[i];
181  }
182 }

186 template <>
187 std::vector<long double> QGaussLobatto<1>::
188 compute_quadrature_points(const unsigned int q,
189  const int alpha,
190  const int beta) const
191 {
192  const unsigned int m = q-2; // no. of inner points
193  std::vector<long double> x(m);
194 
195  // compute quadrature points with
196  // a Newton algorithm.
197 
198  // Set tolerance. See class QGauss
199  // for detailed explanation.
200  const long double
201  long_double_eps = static_cast<long double>(std::numeric_limits<long double>::epsilon()),
202  double_eps = static_cast<long double>(std::numeric_limits<double>::epsilon());
203 
204  // check whether long double is
205  // more accurate than double, and
206  // set tolerances accordingly
207  volatile long double runtime_one = 1.0;
208  const long double tolerance
209  = (runtime_one + long_double_eps != runtime_one
210  ?
211  std::max (double_eps / 100, long_double_eps * 5)
212  :
213  double_eps * 5
214  );
215 
216  // The following implementation
217  // follows closely the one given in
218  // the appendix of the book by
219  // Karniadakis and Sherwin:
220  // Spectral/hp element methods for
221  // computational fluid dynamics
222  // (Oxford University Press, 2005)
223 
224  // we take the zeros of the Chebyshev
225  // polynomial (alpha=beta=-0.5) as
226  // initial values:
227  for (unsigned int i=0; i<m; ++i)
228  x[i] = - std::cos( (long double) (2*i+1)/(2*m) * numbers::PI );
229 
230  long double s, J_x, f, delta;
231 
232  for (unsigned int k=0; k<m; ++k)
233  {
234  long double r = x[k];
235  if (k>0)
236  r = (r + x[k-1])/2;
237 
238  do
239  {
240  s = 0.;
241  for (unsigned int i=0; i<k; ++i)
242  s += 1./(r - x[i]);
243 
244  J_x = 0.5*(alpha + beta + m + 1)*JacobiP(r, alpha+1, beta+1, m-1);
245  f = JacobiP(r, alpha, beta, m);
246  delta = f/(f*s- J_x);
247  r += delta;
248  }
249  while (std::fabs(delta) >= tolerance);
250 
251  x[k] = r;
252  } // for
253 
254  // add boundary points:
255  x.insert(x.begin(), -1.L);
256  x.push_back(+1.L);
257 
258  return x;
259 }

263 template <>
264 std::vector<long double> QGaussLobatto<1>::
265 compute_quadrature_weights(const std::vector<long double> &x,
266  const int alpha,
267  const int beta) const
268 {
269  const unsigned int q = x.size();
270  std::vector<long double> w(q);
271  long double s = 0.L;
272 
273  const long double factor = std::pow(2., alpha+beta+1) *
274  gamma(alpha+q) *
275  gamma(beta+q) /
276  ((q-1)*gamma(q)*gamma(alpha+beta+q+1));
277  for (unsigned int i=0; i<q; ++i)
278  {
279  s = JacobiP(x[i], alpha, beta, q-1);
280  w[i] = factor/(s*s);
281  }
282  w[0] *= (beta + 1);
283  w[q-1] *= (alpha + 1);
284 
285  return w;
286 }

290 template <>
291 long double QGaussLobatto<1>::JacobiP(const long double x,
292  const int alpha,
293  const int beta,
294  const unsigned int n) const
295 {
296  // the Jacobi polynomial is evaluated
297  // using a recursion formula.
298  std::vector<long double> p(n+1);
299 
300  // initial values P_0(x), P_1(x):
301  p[0] = 1.0L;
302  if (n==0) return p[0];
303  p[1] = ((alpha+beta+2)*x + (alpha-beta))/2;
304  if (n==1) return p[1];
305 
306  for (unsigned int i=1; i<=(n-1); ++i)
307  {
308  const int v = 2*i + alpha + beta;
309  const int a1 = 2*(i+1)*(i + alpha + beta + 1)*v;
310  const int a2 = (v + 1)*(alpha*alpha - beta*beta);
311  const int a3 = v*(v + 1)*(v + 2);
312  const int a4 = 2*(i+alpha)*(i+beta)*(v + 2);
313 
314  p[i+1] = static_cast<long double>( (a2 + a3*x)*p[i] - a4*p[i-1])/a1;
315  } // for
316  return p[n];
317 }

321 template <>
322 long double QGaussLobatto<1>::gamma(const unsigned int n)
323 {
324  long double result = n - 1;
325  for (int i=n-2; i>1; --i)
326  result *= i;
327  return result;

412 template <>
413 QGaussLog<1>::QGaussLog(const unsigned int n,
414  const bool revert)
415  :
416  Quadrature<1> (n)
417 {
418  std::vector<double> p = get_quadrature_points(n);
419  std::vector<double> w = get_quadrature_weights(n);
420 
421  for (unsigned int i=0; i<this->size(); ++i)
422  {
423  // Using the change of variables x=1-t, it's possible to show
424  // that int f(x)ln|1-x| = int f(1-t) ln|t|, which implies that
425  // we can use this quadrature formula also with weight ln|1-x|.
426  this->quadrature_points[i] = revert ? Point<1>(1-p[n-1-i]) : Point<1>(p[i]);
427  this->weights[i] = revert ? w[n-1-i] : w[i];
428  }
429 }

696 template<>
697 QGaussLogR<1>::QGaussLogR(const unsigned int n,
698  const Point<1> origin,
699  const double alpha,
700  const bool factor_out_singularity) :
701  Quadrature<1>( ( (origin[0] == 0) || (origin[0] == 1) ) ?
702  (alpha == 1 ? n : 2*n ) : 4*n ),
703  fraction( ( (origin[0] == 0) || (origin[0] == 1.) ) ? 1. : origin[0] )
704 {
705  // The three quadrature formulas that make this one up. There are
706  // at most two when the origin is one of the extremes, and there is
707  // only one if the origin is one of the extremes and alpha is
708  // equal to one.
709  //
710  // If alpha is different from one, then we need a correction which
711  // is performed with a standard Gauss quadrature rule on each
712  // segment. This is not needed in the standard case where alpha is
713  // equal to one and the origin is on one of the extremes. We
714  // integrate with weight ln|(x-o)/alpha|. In the easy cases, we
715  // only need n quadrature points. In the most difficult one, we
716  // need 2*n points for the first segment, and 2*n points for the
717  // second segment.
718  QGaussLog<1> quad1(n, origin[0] != 0);
719  QGaussLog<1> quad2(n);
720  QGauss<1> quad(n);
721 
722  // Check that the origin is inside 0,1
723  Assert( (fraction >= 0) && (fraction <= 1),
724  ExcMessage("Origin is outside [0,1]."));
725 
726  // Non singular offset. This is the start of non singular quad
727  // points.
728  unsigned int ns_offset = (fraction == 1) ? n : 2*n;
729 
730  for (unsigned int i=0, j=ns_offset; i<n; ++i, ++j)
731  {
732  // The first i quadrature points are the same as quad1, and
733  // are by default singular.
734  this->quadrature_points[i] = quad1.point(i)*fraction;
735  this->weights[i] = quad1.weight(i)*fraction;
736 
737  // We need to scale with -log|fraction*alpha|
738  if ( (alpha != 1) || (fraction != 1) )
739  {
740  this->quadrature_points[j] = quad.point(i)*fraction;
741  this->weights[j] = -std::log(alpha/fraction)*quad.weight(i)*fraction;
742  }
743  // In case we need the second quadrature as well, do it now.
744  if (fraction != 1)
745  {
746  this->quadrature_points[i+n] = quad2.point(i)*(1-fraction)+Point<1>(fraction);
747  this->weights[i+n] = quad2.weight(i)*(1-fraction);
748 
749  // We need to scale with -log|fraction*alpha|
750  this->quadrature_points[j+n] = quad.point(i)*(1-fraction)+Point<1>(fraction);
751  this->weights[j+n] = -std::log(alpha/(1-fraction))*quad.weight(i)*(1-fraction);
752  }
753  }
754  if (factor_out_singularity == true)
755  for (unsigned int i=0; i<size(); ++i)
756  {
757  Assert( this->quadrature_points[i] != origin,
758  ExcMessage("The singularity cannot be on a Gauss point of the same order!") );
759  double denominator = std::log(std::abs( (this->quadrature_points[i]-origin)[0] )/alpha);
760  Assert( denominator != 0.0,
761  ExcMessage("The quadrature formula you are using does not allow to "
762  "factor out the singularity, which is zero at one point."));
763  this->weights[i] /= denominator;
764  }
765 }

787 template<>
788 QGaussOneOverR<2>::QGaussOneOverR(const unsigned int n,
789  const Point<2> singularity,
790  const bool factor_out_singularity) :
791  Quadrature<2>(quad_size(singularity, n))
792 {
793  // We treat all the cases in the
794  // same way. Split the element in 4
795  // pieces, measure the area, if
796  // it's relevant, add the
797  // quadrature connected to that
798  // singularity.
799  std::vector<QGaussOneOverR<2> > quads;
800  std::vector<Point<2> > origins;
801  // Id of the corner with a
802  // singularity
803  quads.emplace_back(n, 3, factor_out_singularity);
804  quads.emplace_back(n, 2, factor_out_singularity);
805  quads.emplace_back(n, 1, factor_out_singularity);
806  quads.emplace_back(n, 0, factor_out_singularity);
807 
808  origins.emplace_back(0., 0.);
809  origins.emplace_back(singularity[0], 0.);
810  origins.emplace_back(0., singularity[1]);
811  origins.push_back(singularity);
812 
813  // Lexicographical ordering.
814 
815  double eps = 1e-8;
816  unsigned int q_id = 0; // Current quad point index.
817  Tensor<1,2> dist;
818 
819  for (unsigned int box=0; box<4; ++box)
820  {
821  dist = (singularity-GeometryInfo<2>::unit_cell_vertex(box));
822  dist = Point<2>(std::abs(dist[0]), std::abs(dist[1]));
823  double area = dist[0]*dist[1];
824  if (area > eps)
825  for (unsigned int q=0; q<quads[box].size(); ++q, ++q_id)
826  {
827  const Point<2> &qp = quads[box].point(q);
828  this->quadrature_points[q_id] =
829  origins[box]+
830  Point<2>(dist[0]*qp[0], dist[1]*qp[1]);
831  this->weights[q_id] = quads[box].weight(q)*area;
832  }
833  }

837 template<>
838 QGaussOneOverR<2>::QGaussOneOverR(const unsigned int n,
839  const unsigned int vertex_index,
840  const bool factor_out_singularity) :
841  Quadrature<2>(2*n *n)
842 {
843  // This version of the constructor
844  // works only for the 4
845  // vertices. If you need a more
846  // general one, you should use the
847  // one with the Point<2> in the
848  // constructor.
849  Assert(vertex_index <4, ExcIndexRange(vertex_index, 0, 4));
850 
851  // Start with the gauss quadrature formula on the (u,v) reference
852  // element.
853  QGauss<2> gauss(n);
854 
855  Assert(gauss.size() == n*n, ExcInternalError());
856 
857  // For the moment we only implemented this for the vertices of a
858  // quadrilateral. We are planning to do this also for the support
859  // points of arbitrary FE_Q elements, to allow the use of this
860  // class in boundary element programs with higher order mappings.
861  Assert(vertex_index < 4, ExcIndexRange(vertex_index, 0, 4));
862 
863  // We create only the first one. All other pieces are rotation of
864  // this one.
865  // In this case the transformation is
866  //
867  // (x,y) = (u, u tan(pi/4 v))
868  //
869  // with Jacobian
870  //
871  // J = pi/4 R / cos(pi/4 v)
872  //
873  // And we get rid of R to take into account the singularity,
874  // unless specified differently in the constructor.
875  std::vector<Point<2> > &ps = this->quadrature_points;
876  std::vector<double> &ws = this->weights;
877  double pi4 = numbers::PI/4;
878 
879  for (unsigned int q=0; q<gauss.size(); ++q)
880  {
881  const Point<2> &gp = gauss.point(q);
882  ps[q][0] = gp[0];
883  ps[q][1] = gp[0] * std::tan(pi4*gp[1]);
884  ws[q] = gauss.weight(q)*pi4/std::cos(pi4 *gp[1]);
885  if (factor_out_singularity)
886  ws[q] *= (ps[q]-GeometryInfo<2>::unit_cell_vertex(0)).norm();
887  // The other half of the quadrilateral is symmetric with
888  // respect to xy plane.
889  ws[gauss.size()+q] = ws[q];
890  ps[gauss.size()+q][0] = ps[q][1];
891  ps[gauss.size()+q][1] = ps[q][0];
892  }
893 
894  // Now we distribute these vertices in the correct manner
895  double theta = 0;
896  switch (vertex_index)
897  {
898  case 0:
899  theta = 0;
900  break;
901  case 1:
902  //
903  theta = numbers::PI/2;
904  break;
905  case 2:
906  theta = -numbers::PI/2;
907  break;
908  case 3:
909  theta = numbers::PI;
910  break;
911  }
912 
913  double R00 = std::cos(theta), R01 = -std::sin(theta);
914  double R10 = std::sin(theta), R11 = std::cos(theta);
915 
916  if (vertex_index != 0)
917  for (unsigned int q=0; q<size(); ++q)
918  {
919  double x = ps[q][0]-.5, y = ps[q][1]-.5;
920 
921  ps[q][0] = R00*x + R01*y + .5;
922  ps[q][1] = R10*x + R11*y + .5;
923  }
924 }

1049 QTelles<1>::QTelles (
1050  const Quadrature<1> &base_quad, const Point<1> &singularity)
1051  :
1055  Quadrature<1>(base_quad)
1056 {
1061  const double eta_bar = singularity[0] * 2. - 1.;
1062  const double eta_star = eta_bar * eta_bar - 1.;
1063  double gamma_bar;
1064 
1065  std::vector<Point<1> > quadrature_points_dummy(quadrature_points.size());
1066  std::vector<double> weights_dummy(weights.size());
1067  unsigned int cont = 0;
1068  const double tol = 1e-10;
1069  for (unsigned int d = 0; d < quadrature_points.size(); ++d)
1070  {
1071  if (std::abs(quadrature_points[d][0] - singularity[0]) > tol)
1072  {
1073  quadrature_points_dummy[d-cont] = quadrature_points[d];
1074  weights_dummy[d-cont] = weights[d];
1075  }
1076  else
1077  {
1078  // We need to remove the singularity point from the quadrature point
1079  // list. To do so we use the variable cont.
1080  cont = 1;
1081  }
1082 
1083  }
1084  if (cont == 1)
1085  {
1086  quadrature_points.resize(quadrature_points_dummy.size()-1);
1087  weights.resize(weights_dummy.size()-1);
1088  for (unsigned int d = 0; d < quadrature_points.size()-1; ++d)
1089  {
1090  quadrature_points[d] = quadrature_points_dummy[d];
1091  weights[d] = weights_dummy[d];
1092  }
1093  }
1094  // We need to check if the singularity is at the boundary of the interval.
1095  if (std::abs(eta_star) <= tol)
1096  {
1097  gamma_bar = std::pow((eta_bar * eta_star + std::abs(eta_star)),1.0 / 3.0)
1098  + std::pow((eta_bar * eta_star - std::abs(eta_star)), 1.0 / 3.0)
1099  + eta_bar;
1100  }
1101  else
1102  {
1103  gamma_bar = (eta_bar * eta_star + std::abs(eta_star))/std::abs(eta_bar * eta_star + std::abs(eta_star))*
1104  std::pow(std::abs(eta_bar * eta_star + std::abs(eta_star)),1.0 / 3.0)
1105  + (eta_bar * eta_star - std::abs(eta_star))/std::abs(eta_bar * eta_star - std::abs(eta_star))*
1106  std::pow(std::abs(eta_bar * eta_star - std::abs(eta_star)), 1.0 / 3.0)
1107  + eta_bar;
1108  }
1109  for (unsigned int q = 0; q < quadrature_points.size(); ++q)
1110  {
1111  double gamma = quadrature_points[q][0] * 2 - 1;
1112  double eta = (std::pow(gamma - gamma_bar, 3.0)
1113  + gamma_bar * (gamma_bar * gamma_bar + 3))
1114  / (1 + 3 * gamma_bar * gamma_bar);
1115 
1116  double J = 3 * ((gamma - gamma_bar) *(gamma - gamma_bar))
1117  / (1 + 3 * gamma_bar * gamma_bar);
1118 
1119  quadrature_points[q][0] = (eta + 1) / 2.0;
1120  weights[q] = J * weights[q];
1121 
1122  }
