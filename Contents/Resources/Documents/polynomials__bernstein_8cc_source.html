28  template <typename number>
29  std::vector<number>
30  get_bernstein_coefficients (
31  const unsigned int k, const unsigned int n)
32  {
33  Assert(n>0, ExcMessage("Bernstein polynomial needs to be of degree > 0."));
34  AssertIndexRange(k, n+1);
35  std::vector<number> coeff(n + 1, number(0.0));
36  for (unsigned int i = k; i < n + 1; ++i)
37  {
38  coeff[i] = ((i - k) % 2 == 0 ? 1 : -1)
39  * boost::math::binomial_coefficient<number>(n, i)
40  * boost::math::binomial_coefficient<number>(i, k);
41  }
42  return coeff;
43  }
44 }
